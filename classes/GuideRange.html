50 def self.new_from_eol_data_object(data_object)
51   thumb_url = data_object.at('thumbnailURL').try(:content)
52   p = /_\d+_\d+\.(\w+)$/
53   original_url = if thumb_url && thumb_url.match(p)
54     thumb_url.sub(p, "_orig.\\1")
55   else
56     data_object.at('mediaURL').try(:content)
57   end
58   medium_url = if thumb_url && thumb_url.match(p)
59     thumb_url.sub(p, "_580_360.\\1")
60   else
61     original_url
62   end
63   rights_holder = data_object.at('rightsHolder').try(:content)
64   rights_holder ||= data_object.at('agent[role=compiler]').try(:content)
65   rights_holder ||= data_object.at('agent[role=author]').try(:content)
66   gr = GuideRange.new(
67     :rights_holder => rights_holder,
68     :thumb_url => thumb_url,
69     :medium_url => medium_url,
70     :original_url => original_url
71   )
72   gr.license = case data_object.at('license').to_s
73   when /\/by\// then Observation::CC_BY
74   when /\/by-nc\// then Observation::CC_BY_NC
75   when /\/by-sa\// then Observation::CC_BY_SA
76   when /\/by-nd\// then Observation::CC_BY_ND
77   when /\/by-nc-sa\// then Observation::CC_BY_NC_SA
78   when /\/by-nc-nd\// then Observation::CC_BY_NC_ND
79   when /\/publicdomain\// then Photo::PD_CODE
80   else
81     data_object.at('license').to_s
82   end
83   gr.source_url = if (verson_id = data_object.at('dataObjectVersionID').try(:content))
84     "http://eol.org/data_objects/#{verson_id}"
85   elsif data_object.at('source').to_s =~ /http/
86     data_object.at('source').to_s
87   end
