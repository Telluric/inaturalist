289 def autocomplete
290   @q = params[:q] || params[:term] || params[:item]
291   @q = sanitize_query(@q.to_s.sanitize_encoding)
292   site_place = @site.place if @site
293   params[:per_page] ||= 30
294   if @q.blank?
295     scope = if site_place
296       Place.where(site_place.child_conditions)
297     else
298       Place.where("place_type = ?", Place::CONTINENT).order("updated_at desc")
299     end
300     scope = scope.with_geom if params[:with_geom]
301     scope = scope.with_check_list if params[:with_check_list]
302     @places = scope.includes(:place_geometry_without_geom).limit(params[:per_page]).
303       sort_by{|p| p.bbox_area || 0}.reverse
304   else
305     # search both the autocomplete and normal field
306     # autocomplete doesn't work well with 1- or 2-letter words
307     filters = [ { bool: { should: [
308       { match: { display_name_autocomplete: @q } },
309       { match: { display_name: { query: @q, operator: "and" } } }
310     ] } } ]
311     inverse_filters = []
312     if site_place
313       filters << { term: { ancestor_place_ids: site_place.id } }
314     end
315     if params[:with_geom]
316       filters << { exists: { field: :geometry_geojson } }
317     end
318     if params[:with_check_list]
319       inverse_filters << { exists: { field: :without_check_list } }
320     end
321     @places = Place.elastic_paginate(
322       filters: filters,
323       inverse_filters: inverse_filters,
324       sort: { bbox_area: "desc" },
325       per_page: params[:per_page])
326     Place.preload_associations(@places, :place_geometry_without_geom)
327   end
328 
329   respond_to do |format|
330     format.html do
331       render :layout => false, :partial => 'autocomplete' 
332     end
333     format.json do
334       @places.each_with_index do |place, i|
335         @places[i].html = view_context.render_in_format(:html, :partial => 'places/autocomplete_item', :object => place)
336       end
337       render json: @places.to_json(methods: [:html, :kml_url]),
338         callback: params[:callback]
