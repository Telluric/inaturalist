Class PlaceGeometry < ApplicationRecord

  • app/models/place_geometry.rb

 80         SELECT
 81           ST_RemoveRepeatedPoints(ST_Multi(ST_Union(ST_MakeValid(geom)))) AS new_geom

 83           SELECT (ST_Dump(geom)).geom

 89     SQL
 90   rescue StandardError => e
 91     raise unless e.message =~ /conversion failed/

 93       Rails.logger.error "[ERROR #{Time.now}] Failed to dissolve for PlaceGeometry #{id}, attempting to simplify: #{e}"
 94       connection.execute <<-SQL
 95         UPDATE place_geometries SET geom = reuonioned_geoms.new_geom FROM (
 96           SELECT
 97             ST_RemoveRepeatedPoints(ST_Multi(ST_Union(geom))) AS new_geom
 98           FROM (
 99             SELECT ST_SimplifyPreserveTopology((ST_Dump(geom)).geom, 0.0001) AS geom
100             FROM place_geometries

102           ) AS expanded_geoms
103         ) AS reuonioned_geoms
104         WHERE id = #{id}
105       SQL

109       Rails.logger.error "[ERROR #{Time.now}] Failed to dissolve for PlaceGeometry #{id}, filtering invalid geoms: #{e}"
110       connection.execute <<-SQL
111         UPDATE place_geometries SET geom = reuonioned_geoms.new_geom FROM (
112           SELECT
113             ST_RemoveRepeatedPoints(ST_Multi(ST_Union(geom))) AS new_geom

148 def simplified_geom
149   return if !geom
150   if !place.bbox_area || place.bbox_area < 0.1
151     # this method is currently only used for indexing places in Elasticsearch.
152     # Running the cleangeometry method here helps fix geom validation errors
153     # which psql is comfortable with but might cause ES to throw errors
154     return PlaceGeometry.where(id: id).
155       select("id, cleangeometry(geom) as simpl").first.try(:simpl)
156   end
157   tolerance =
158     if place.bbox_area < 1
159       0.001
160     elsif place.bbox_area < 10
161       0.01
162     elsif place.bbox_area < 100
163       0.017
164     elsif place.bbox_area < 500
165       0.04
166     else
167       0.075
168     end
169   if s = PlaceGeometry.where(id: id).
170     select("id, cleangeometry(ST_SimplifyPreserveTopology(geom, #{ tolerance })) as simpl").first.simpl rescue nil
171     return s
172   end
173   PlaceGeometry.where(id: id).
174     select("id, cleangeometry(ST_Buffer(ST_SimplifyPreserveTopology(geom, #{ tolerance }),0)) as simpl").first.simpl
175 end

25 def validate_geometry
26   # not sure why this is necessary, but validates_presence_of :geom doesn't always seem to run first
27   if geom.blank?
28     errors.add(:geom, :cannot_be_blank)
29     return
30   end
31   if geom.num_points < 4
32     errors.add(:geom, :must_have_more_than_three_points)
33   end
34   if geom.detect{|g| g.num_points < 4}
35     errors.add(:geom, :polygon_with_less_than_four_points)
36   end
37   if geom.detect{|g| g.points.detect{|pt| pt.x < -180 || pt.x > 180 || pt.y < -90 || pt.y > 90}}
38     errors.add(:geom, :invalid_point)
39   end
40 end
