56 def get_uri(method, *args)
57   arg = args.first unless args.first.is_a?(Hash)
58   params = args.detect{|a| a.is_a?(Hash)} || {}
59   uri = "#{ api_endpoint.base_url }#{ method }/#{ SERVICE_VERSION }"
60   uri += "/#{arg}" if arg
61   uri += ".xml"
62   unless params.blank?
63     uri += "?"
64     uri += params.map {|k,v| "#{k}=#{v}"}.join('&') 
65   end
66   URI.parse(URI.encode(uri))
67 end
