   # File lib/cloudfront_invalidator.rb
 5 def invalidate_cloudfront_cache_for(attachment_name, path_pattern)
 6   attr_to_check = "#{attachment_name}_updated_at"
 7   # the attachment has been modified, and was an attachment before
 8   return if try(:skip_cloudfront_invalidation)
 9   if respond_to?( :in_public_s3_bucket? ) && in_public_s3_bucket?
10     # the public bucket does not use cloudfront
11     return
12   end
13   if saved_changes[attr_to_check] && !saved_changes[attr_to_check][0].nil?
14     begin
15       INatAWS.cloudfront_invalidate(path_pattern.sub(":id", id.to_s))
16     rescue Exception => e
17       Rails.logger.error "[ERROR #{Time.now}] Failed CloudFront invalidation: #{e}"
18     end
19   end
20 end
