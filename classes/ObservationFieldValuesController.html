109 def destroy
110   errors = []
111   if @observation_field_value.blank?
112     status = :gone
113     errors << "Observation field value #{params[:id]} does not exist."
114   elsif @observation_field_value.observation.user_id != current_user.id &&
115         @observation_field_value.user_id != current_user.id
116     status = :forbidden
117     errors << t(:you_dont_have_permission_to_do_that)
118   else
119     proj_requiring_field = @observation_field_value.observation.projects.detect do |proj|
120       proj.project_observation_fields.detect do |pof|
121         pof.required? && pof.observation_field_id === @observation_field_value.observation_field_id
122       end
123     end
124     if proj_requiring_field
125       status = :unprocessable_entity
126       errors << t(:observation_belongs_to_project_requiring_field)
127     else
128       @observation_field_value.destroy
129       status = :ok
130     end
131   end
132   respond_to do |format|
133     format.any do
134       render status: status, plain: errors ? errors.join( ", " ) : nil
135     end
136     format.json do 
137       render status: status, json: errors ? { errors: errors } : nil
138     end
139   end
