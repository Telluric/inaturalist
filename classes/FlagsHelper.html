58 def flag_resolution( flag, site )
59   return unless flag.resolved?
60 
61   resolved_at = flag.resolved_at || flag.updated_at
62   resolved_by = if flag.resolver
63     link_to_user( flag.resolver )
64   elsif flag.resolver_id
65     t :deleted_user
66   else
67     site.name
68   end
69 
70   resolver = content_tag :div, class: "text-muted" do
71     t :resolved_by_user_on_date_html, user: resolved_by, date: resolved_at ? l( resolved_at ) : t( :unknown )
72   end
73 
74   [formatted_user_text( flag.comment ), resolver].compact.join.html_safe
75 end

77 def flaggable_edit( flaggable )
78   return "" unless editable?( flaggable )
79 
80   edit_path = if flaggable.is_a?( Sound )
81     sound_path( flaggable )
82   elsif flaggable.is_a?( Photo )
83     photo_path( flaggable )
84   elsif respond_to?( "edit_#{flaggable.class.name.underscore}_path" )
85     send( "edit_#{flaggable.class.name.underscore}_path", flaggable )
86   end
87   return "" if edit_path.blank?
88 
89   link_to( edit_path ) do
90     content_tag :i, "", class: "fa fa-pencil"
91   end
92 end
