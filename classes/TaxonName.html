278 def self.choose_common_name( taxon_names, options = {} )
279   return nil if taxon_names.blank?
280   if options[:user] && !options[:user].prefers_common_names?
281     return nil
282   end
283 
284   common_names = taxon_names.reject {| tn | tn.is_scientific_names? || !tn.is_valid? }
285   return nil if common_names.blank?
286 
287   place_id = options[:place_id] unless options[:place_id].blank?
288   place_id ||= ( options[:place].is_a?( Place ) ? options[:place].id : options[:place] ) unless options[:place].blank?
289   place_id ||= options[:user].place_id unless options[:user].blank?
290 
291   if place_id.blank? && options[:site] && !options[:site].place_id.blank?
292     place_id ||= options[:site].place_id
293   end
294   place = ( options[:place].is_a?( Place ) ? options[:place] : Place.find_by_id( place_id ) ) unless place_id.blank?
295   common_names = common_names.sort_by {| tn | [tn.position, tn.id] }
296 
297   if place
298     exact_place_names = common_names.select {| tn | tn.place_taxon_names.detect {| ptn | ptn.place_id == place.id } }
299     place_names = []
300     if exact_place_names.blank?
301       place_names = common_names.select do | tn |
302         tn.place_taxon_names.detect do | ptn |
303           place.self_and_ancestor_ids.include?( ptn.place_id )
304         end
305       end
306     end
307     name_sorter = proc do | tn |
308       ptn = tn.place_taxon_names.detect {| ptn1 | ptn1.place_id == place.id }
309       ptn ||= tn.place_taxon_names.detect {| ptn2 | place.self_and_ancestor_ids.include?( ptn2.place_id ) }
310       [ptn.position, tn.position, tn.id]
311     end
312     place_names = place_names.sort_by( &name_sorter )
313     exact_place_names = exact_place_names.sort_by( &name_sorter )
314   else
315     place_names = []
316     exact_place_names = []
317   end
318   locale = options[:locale]
319   locale = options[:user].try( :locale ) if locale.blank?
320   locale = options[:site].try( :locale ) if locale.blank?
321   locale = I18n.locale if locale.blank?
322   language_name = language_for_locale( locale )
323   locale_names = common_names.select {| n | n.localizable_lexicon == language_name }
324 
325   # We want Maori names to show up in New Zealand even for English speakers,
326   # but we don't want North American English names to show in Mexcio
327   locale_and_place_names = place_names.select {| n | n.localizable_lexicon == language_name }
328   exact_locale_and_place_names = exact_place_names.select {| n | n.localizable_lexicon == language_name }
329 
330   if exact_locale_and_place_names.length.positive?
331     exact_locale_and_place_names.first
332   elsif exact_place_names.length.positive?
333     exact_place_names.first
334   elsif locale_and_place_names.length.positive?
335     locale_and_place_names.first
336   elsif locale_names.length.positive?
337     locale_names.first
338   end
339 end

416 def self.find_external( query, options = {} )
417   r = ratatosk( options )
418   # fetch names and save them
419   ext_names = r.find( query )
420   existing = Taxon.where( name: ext_names.map( &:name ) ).limit( 500 ).index_by( &:name )
421   ext_names.map do | ext_name |
422     if !ext_name.valid? && ( existing_taxon = r.find_existing_taxon( ext_name.taxon ) )
423       ext_name.taxon = existing_taxon
424     end
425     if existing[ext_name.name]
426       # don't bother creating new synonymous taxa
427       nil
428     elsif ext_name.save
429       ext_name
430     else
431       Rails.logger.debug "[DEBUG] Failed to save ext_name: #{ext_name.errors.full_messages.to_sentence}"
432       nil
433     end
434   end.compact
435   ext_names.each do | ext_name |
436     matching_tn_exists = ext_name.taxon.taxon_names.detect do | tn |
437       tn.is_valid? && tn.name == ext_name.taxon.name
438     end
439     if !ext_name.valid? && ext_name.errors[:name] && matching_tn_exists
440       ext_name.update( is_valid: false )
441     end
442   end
443   ext_names
444 end

446 def self.find_single( name, options = {} )
447   lexicon = options.delete( :lexicon ) # || TaxonName::LEXICONS[:SCIENTIFIC_NAMES]
448   name = name.split( "_" ).join( " " )
449   name = Taxon.remove_rank_from_name( name )
450   conditions = { name: name }
451   conditions[:lexicon] = lexicon if lexicon
452   begin
453     taxon_names = TaxonName.where( conditions ).limit( 10 ).includes( :taxon )
454   rescue ActiveRecord::StatementInvalid => e
455     raise e unless e.message =~ /invalid byte sequence/
456 
457     conditions[:name] = name.encode( "UTF-8" )
458     taxon_names = TaxonName.where( conditions ).limit( 10 ).includes( :taxon )
459   end
460   unless options[:iconic_taxa].blank?
461     taxon_names.reject {| tn | options[:iconic_taxa].include?( tn.taxon.iconic_taxon_id ) }
462   end

225 def self.normalize_lexicon( lexicon )
226   return nil if lexicon.blank?
227   return TaxonName::NORWEGIAN if lexicon == "norwegian_bokmal"
228   return TaxonName::PERSIAN if lexicon.to_s.downcase.strip == "farsi"
229   return TaxonName::ROMANIAN if lexicon.to_s.downcase.strip == "rumanian"
230   return TaxonName::SWAHILI if lexicon.to_s.downcase.strip == "kiswahili"
231   return TaxonName::SWAHILI if lexicon.to_s =~ /^swahili/
232   # Correct a common misspelling
233   return TaxonName::UKRAINIAN if lexicon.to_s.downcase.strip == "ukranian"
234 
235   ( LEXICONS[lexicon.underscore.upcase.to_sym] || lexicon.titleize ).strip
236 end
