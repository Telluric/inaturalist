245         puts msg
246       end
247     end
248   end
249

70     flatten.compact.uniq

77   else
78     places = places.where( "( places.admin_level = ? OR places.id IN ( ? ) )", Place::COUNTRY_LEVEL, exploded_place_ids_to_include )

80   # Exclude places if necessary
