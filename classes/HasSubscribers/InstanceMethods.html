349 def delete_update_actions
350   # some classes, like ListedTaxa, have tons of records but barely any UpdateActions or
351   # Subscriptions. Check to make sure there are some subscriptions for this record before

354     UpdateAction.delete_and_purge(["resource_type = ? AND resource_id = ?", self.class.base_class.name, self.id])

321 def notify_users( method )
322   return if CONFIG.has_subscribers == :disabled
323   options = self.class.notifies_users_options
324   resource = observation if respond_to?(:observation)
325   resource = parent if is_a?( Comment )
326   resource ||= self
327   action_attrs = {
328     resource: resource,
329     notifier: self,
330     notification: options[:notification]
331   }
332   users = send(method)
333   except_users = send( options[:except].to_sym ) unless options[:except].blank?
334   except_users ||= []
335   if users.empty?
336     UpdateAction.delete_and_purge(action_attrs)
337     return
338   end
339   users_to_notify = users - except_users
340   user_ids_to_notify = users_to_notify.map{ |u|
341     options[:if].blank? || options[:if].call( u ) ? u.id : nil
342   }.compact
343   if action = UpdateAction.first_with_attributes(action_attrs)
344     action.append_subscribers( user_ids_to_notify )
345     action.restrict_to_subscribers( user_ids_to_notify )
346   end
347 end
