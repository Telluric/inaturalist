 73 def self.fetch_request_uri(options = {})
 74   return unless options[:request_uri]
 75   options[:timeout] ||= 5
 76   options[:user_agent] ||= Site.default.name
 77   if options[:api_endpoint]
 78     api_endpoint_cache = ApiEndpointCache.find_or_create_by(
 79       api_endpoint: options[:api_endpoint],
 80       request_url: options[:request_uri].to_s)
 81     return if api_endpoint_cache.in_progress?
 82     if api_endpoint_cache.cached? && !options[:force_update]
 83       return Nokogiri::XML(api_endpoint_cache.response)
 84     end
 85   end
 86   response = nil
 87   begin
 88     if api_endpoint_cache
 89       api_endpoint_cache.update(request_began_at: Time.now,
 90         request_completed_at: nil, success: nil, response: nil)
 91     end
 92     timed_out = Timeout::timeout(options[:timeout]) do
 93       response = fetch_with_redirects(options)
 94     end
 95   rescue Timeout::Error
 96     if api_endpoint_cache
 97       api_endpoint_cache.update(
 98         request_completed_at: Time.now, success: false)
 99     end
100     raise Timeout::Error
101   end

104       request_completed_at: Time.now,
105       success: !response.body.blank?,

107     )
108   end
109   Nokogiri::XML(response.body)

112 def self.fetch_with_redirects(options, attempts = 3)
113   http = Net::HTTP.new(options[:request_uri].host, options[:request_uri].port)
114   # using SSL if we have an https URL
115   http.use_ssl = (options[:request_uri].scheme == "https")
116   response = http.get("#{options[:request_uri].path}?#{options[:request_uri].query}",
117     "User-Agent" => options[:user_agent])
118   # following redirects if we haven't followed too many already
119   if response.is_a?(Net::HTTPRedirection) && attempts > 0
120     options[:request_uri] = URI.parse(response["location"])
121     return fetch_with_redirects(options, attempts - 1)
122   end
123   response
124 end
