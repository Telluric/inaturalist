 5 def self.describe(taxon, options = {})
 6   if options[:describer]
 7     txt = case options[:describer].to_s.downcase
 8     when "amphibiaweb" then TaxonDescribers::AmphibiaWeb.describe(taxon)
 9     when "eol" then TaxonDescribers::Eol.describe(taxon)
10     when "conabio" then TaxonDescribers::Conabio.describe(taxon)
11     end
12     return txt
13   end
14   describers = options[:describers]
15   describers = [Wikipedia, Eol] if describers.blank?
16   describers.each do |describer|
17     text = describer.describe(taxon)
18     return text unless text.blank?
19   end
20 end
