146 def merge_field
147   @keeper = ObservationField.find_by_id(params[:with])
148   @reject = @observation_field
149   error = if @keeper.blank?
150     t(:you_must_choose_an_observation_field)
151   elsif @reject.project_observation_fields.exists?
152     t(:you_cant_merge_observation_fields_in_use_by_projects)
153   end
154   if error
155     respond_to do |format|
156       format.html do
157         flash[:error] = error
158         redirect_back_or_default @observation_field
159       end
160     end
161     return
162   end
163 
164   merge = []
165   keepers = []
166   params.each do |k,v|
167     if v.is_a?(Array) && v.size == 2
168       merge << k.gsub('keep_', '')
169     elsif k =~ /^keep_/ && v == 'reject'
170       keepers << k.gsub('keep_', '')
171     end
172   end
173 
174   @keeper.merge(@reject, :merge => merge, :keep => keepers)
175 
176   respond_to do |format|
177     format.html do
178       flash[:notice] = t(:fields_merged)
179       redirect_to @keeper
180     end
181   end
182 end
