286 def self.arel_attributes_to_es_filters( attrs )
287   filters = []
288   attrs.each do |k,v|
289     # base_class is used because UpdateAction uses polymorphic associations
290     # for resource and notifier which use base_class for setting `*_type`
291     if k.to_sym == :resource
292       filters << { term: { resource_id: v.id } }
293       filters << { term: { resource_type: v.class.base_class.name } }
294     elsif k.to_sym == :notifier
295       filters << { term: { notifier_id: v.id } }
296       filters << { term: { notifier_type: v.class.base_class.name } }
297     else
298       filters << { term: { k => v.try(:id) || v } }
299     end
300   end
301   filters
302 end

257 def self.first_with_attributes(attrs, options = {})
258   return if CONFIG.has_subscribers == :disabled
259   skip_validations = options.delete(:skip_validations)
260   filters = UpdateAction.arel_attributes_to_es_filters( attrs )
261   if action = UpdateAction.elastic_paginate(filters: filters, keep_es_source: true).first
262     return action
263   end
264   begin
265     action = UpdateAction.new(attrs.merge(created_at: Time.now, skip_indexing: true).merge(options))
266     if !action.save(validate: !skip_validations)
267       return
268     end
269     action.created_but_not_indexed = true
270     return action
271   rescue PG::Error, ActiveRecord::RecordNotUnique => e
272     # caught a record not unique error. Try ES once more before returning
273     Logstasher.write_exception(e, reference: "UpdateAction.first_with_attributes RecordNotUnique")
274     sleep( 1 )
275     10.times do |iteration|
276       if action = UpdateAction.elastic_paginate(filters: filters, keep_es_source: true).first
277         return action
278       else
279         sleep( 1 )
280       end

178 def self.group_and_sort(updates, options = {})
179   grouped_updates = []
180   updates.group_by{|u| [u.resource_type, u.resource_id, u.notification]}.each do |key, batch|
181     resource_type, resource_id, notification = key
182     batch = batch.sort_by{|u| u.sort_by_date}
183     if options[:hour_groups] && "created_observations new_observations".include?(notification.to_s) && batch.size > 1
184       batch.group_by{|u| u.created_at.strftime("%Y-%m-%d %H")}.each do |hour, hour_updates|
185         grouped_updates << [key, hour_updates]
186       end
187     elsif notification == "activity" && !options[:skip_past_activity]
188       # get the resource that has all this activity
189       resource = batch.first.resource
190       if resource.blank?
191         Rails.logger.error "[ERROR #{Time.now}] couldn't find resource #{resource_type} #{resource_id}, first update: #{batch.first}"
192         next
193       end
194 
195       # get the associations on that resource that generate activity updates
196       activity_assocs = resource.class.notifying_associations.select do |assoc, assoc_options|
197         assoc_options[:notification] == "activity"
198       end
199 
200       # create pseudo updates for all activity objects
201       activity_assocs.each do |assoc, assoc_options|
202         # this is going to lazy load assoc's of the associate (e.g. a comment's user) which might not be ideal
203         resource.send(assoc).each do |associate|
204           unless batch.detect{|u| u.notifier_type == associate.class.name && u.notifier_id == associate.id}
205             batch << UpdateAction.new(:resource => resource, :notifier => associate, :notification => "activity")
206           end
207         end
208       end
209       grouped_updates << [key, batch.sort_by{|u| u.sort_by_date}]
210     else
211       grouped_updates << [key, batch]
212     end
213   end
214   grouped_updates.sort_by {|key, updates| updates.last.sort_by_date.to_i * -1}
215 end

168   end
