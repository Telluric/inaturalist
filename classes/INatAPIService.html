48 def self.get_json( path, params = {}, options = {} )
49   options[:retries] ||= 3
50   options[:timeout] ||= INatAPIService::TIMEOUT
51   options[:retry_delay] ||= 0.1
52   endpoint = options[:endpoint] || INatAPIService::ENDPOINT
53   url = endpoint + path
54   headers = {}
55   auth_user = params.delete(:authenticate)
56   if auth_user && auth_user.is_a?( User )
57     headers["Authorization"] = auth_user.api_token
58   end
59   authorization = options.delete(:authorization)
60   if authorization && !headers["Authorization"]
61     headers["Authorization"] = authorization
62   end
63   begin
64     uri = URI.parse(url)
65   rescue URI::InvalidURIError
66     uri = URI.parse(URI.escape(url))
67   end
68   if !params.blank? && params.is_a?(Hash)
69     uri.query = URI.encode_www_form( Hash[URI.decode_www_form( uri.query || "" )].merge( params ) )
70   end
71   begin
72     timed_out = Timeout::timeout( options[:timeout] ) do
73       http = Net::HTTP.new(uri.host, uri.port)
74       http.use_ssl = true if uri.scheme == "https"
75       # http.verify_mode = OpenSSL::SSL::VERIFY_NONE
76       response = http.get( uri.request_uri, headers )
77       if response.code == "200"
78         return response.body.force_encoding( "utf-8" )
79       end
80     end
81   rescue => e
82     Rails.logger.debug "[DEBUG] INatAPIService.get_json(#{path}, #{params}, #{options[:retries]}) failed: #{e}"
83   end
84   if options[:retries].is_a?(Integer) && options[:retries] > 0
85     retry_options = options.dup
86     retry_options[:retries] -= 1
87     if options[:retry_delay]
88       # delay a bit before retrying
89       sleep options[:retry_delay]
90     end
91     return INatAPIService.get_json( path, params, retry_options )
92   end
93   false
94 end
