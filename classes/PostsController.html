169 def update

171   @post.published_at = nil if params[:commit] == t(:unpublish)
172   if params[:observations]

174     params[:observations] = ((params[:observations] & @post.observation_ids) + params[:observations]).uniq
175     @observations = Observation.by(current_user).where(id: params[:observations])
176   end

179     @preview = @post

181     return render(action: 'edit')
182   end

184   # This will actually perform the updates / deletions, so it needs to

188   else
