240 def self.repair(find_options = {})
241   puts "[INFO #{Time.now}] starting FlickrPhoto.repair, options: #{find_options.inspect}"
242   find_options[:include] ||= [{:user => :flickr_identity}, :taxon_photos, :observation_photos]
243   find_options[:batch_size] ||= 100
244   find_options[:sleep] ||= 10

246   updated = 0
247   destroyed = 0
248   invalids = 0
249   skipped = 0
250   start_time = Time.now
251   counter = 0
252   FlickrPhoto.includes(find_options[:include]).where(find_options[:where]).
253     find_each(batch_size: find_options[:batch_size]) do |p|
254     counter += 1
255     sleep(find_options[:sleep]) if (counter % find_options[:batch_size] == 0)
256     r = begin
257       uri = URI.parse(p.square_url)
258       Net::HTTP.new(uri.host).request_head(uri.path)
259     rescue URI::InvalidURIError
260       puts "[ERROR] Failed to retrieve #{p.square_url}, skipping..."
261       skipped += 1
262       next
263     rescue Timeout::Error, Errno::ECONNREFUSED => e
264       puts "[ERROR] #{e.message} (#{e.class.name}), skipping..."
265       skipped += 1
266       next
267     end
268     unless r.is_a?(Net::HTTPBadRequest) || r.is_a?(Net::HTTPRedirection)
269       skipped += 1
270       next
271     end
272     repaired, errors = p.repair
273     if errors.blank?
274       updated += 1
275     else
276       puts "[ERROR] #{errors.values.to_sentence}"
277       if repaired.frozen?
278         destroyed += 1 
279         puts "[ERROR] destroyed #{repaired}"
280       end
281       if errors[:flickr_authorization_missing]
282         invalids += 1
283         puts "[ERROR] authorization missing #{repaired}"
284       end
285     end
286   end
287   puts "[INFO #{Time.now}] finished FlickrPhoto.repair, #{updated} updated, #{destroyed} destroyed, #{invalids} invalid, #{skipped} skipped, #{Time.now - start_time}s"

172   tags = to_tags( options )
173   taxa = if tags.blank?
174     []
175   else
