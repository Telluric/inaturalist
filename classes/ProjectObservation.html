242 def to_csv_column(column, options = {})
243   p = options[:project] || project
244   case column
245   when "curator_ident_taxon_id"
246     curator_identification.try(:taxon_id)
247   when "curator_ident_taxon_name"
248     if curator_identification
249       curator_identification.taxon.name
250     else
251       nil
252     end
253   when "curator_ident_user_id"
254     curator_identification.try(:user_id)
255   when "curator_ident_user_login"
256     if curator_identification
257       curator_identification.user.login
258     else
259       nil
260     end
261   when "curator_coordinate_access"
262     preferred_curator_coordinate_access
263   else
264     if column.to_s =~ /private_/
265       if observation && observation.coordinates_viewable_by?(options[:viewer])
266         observation.send(column)
267       else
268         nil
269       end
270     elsif observation_field = p.observation_fields.detect{|of| of.normalized_name == ObservationField.normalize_name( column )}
271       observation.observation_field_values.detect{|ofv| ofv.observation_field_id == observation_field.id}.try(:value)
272     else
273       observation.send(column) rescue send(column) rescue nil
274     end
275   end
276 end
