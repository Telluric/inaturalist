59   @is_active = filter_params[:is_active]
60   
61   scope = TaxonFrameworkRelationship.all
62   scope = scope.relationships(@relationships) unless @relationships.blank?
63   scope = scope.taxon_framework(@taxon_framework) if @taxon_framework
64   scope = scope.internal_taxon(@internal_taxon) if @internal_taxon
65   scope = scope.external_taxon(@external_taxon_name) if @external_taxon_name
66   scope = scope.internal_rank(@internal_rank) if @internal_rank
67   scope = scope.external_rank(@external_rank) if @external_rank
68   scope = scope.by(@user) if @user
69   if @is_active.yesish?
70     scope = scope.active
71   elsif @is_active.noish?
72     scope = scope.inactive
73   end
74   
75   @taxon_framework_relationships = scope.page(params[:page]).
76     select("DISTINCT ON (taxon_framework_relationships.id) taxon_framework_relationships.*").
77     includes(:external_taxa ,taxa: [:taxon_names, :photos], taxon_framework: :taxon).
78     order("taxon_framework_relationships.id DESC")
79 
80   respond_to do |format|
81     format.html
82     format.json do
83       taxon_options = { only: [:id, :name, :rank] }
84       external_taxon_options = { only: [:name, :url] }
85       render json: @taxon_framework_relationships.as_json(
86         methods: [:relationship],
87         include: [
88           { taxa: taxon_options },
89           { external_taxa: external_taxon_options }
90         ]

92     end
