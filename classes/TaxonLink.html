 90 def self.by_taxon( taxon, options = {} )
 91   return [] if taxon.blank?
 92   taxon_links = if taxon.species_or_lower?
 93     # fetch all relevant links
 94     TaxonLink.for_taxon( taxon ).includes( :taxon ).to_a
 95   else
 96     # fetch links without species only
 97     TaxonLink.for_taxon( taxon ).where( species_only: false ).includes( :taxon ).to_a
 98   end
 99   tl_place_ids = taxon_links.map(&:place_id).compact
100   # Trying out removing the place-specific filtering. We already have the
101   # checklist filtering, so maybe this doesn't matter. Queue the screaming
102   # hordes...
103   # if options[:place]
104   #   tl_place_ids = options[:place].self_and_ancestor_ids
105   # end
106   if !tl_place_ids.blank?
107     if options[:reject_places]
108       taxon_links.reject! { |tl| tl.place_id }
109     else
110       # fetch listed taxa for this taxon with places matching the links
111       place_listed_taxa = ListedTaxon.where( "place_id IN (?)", tl_place_ids ).where( taxon_id: taxon )
112 
113       # remove links that have a place_id set but don't have a corresponding listed taxon
114       taxon_links.reject! do |tl|
115         tl.place_id && place_listed_taxa.detect { |lt| lt.place_id == tl.place_id }.blank?
116       end
117     end
118   end
119   taxon_links.uniq! { |tl| tl.url }
120   taxon_links.sort_by { |tl| tl.taxon.ancestry || "" }.reverse
121 end

52 def url_for_taxon(taxon)
53   new_url = url.sub( "[NAME]", taxon.name )
54   new_url = new_url.sub( "[NAME_]", taxon.name.strip.gsub( /\s+/, "_" ) )
55   new_url = new_url.sub( "[NAME-]", taxon.name.strip.gsub( /\s+/, "-" ) )
56   new_url = new_url.sub( "[RANK]", taxon.rank )
57   new_url = if taxon.rank_level.to_i < Taxon::SPECIES_LEVEL
58     new_url.sub( "[NAME_WITH_RANK]", taxon.name_with_rank )
59   else
60     new_url.sub( "[NAME_WITH_RANK]", taxon.name )
61   end
62   if taxon.species_or_lower? && pieces = taxon.name.split
63     new_url.sub!( "[GENUS]", pieces.first )
64     new_url.sub!( "[SPECIES]", pieces[1] || "" )
65   else
66     new_url.sub!( /\[GENUS\].*\[SPECIES\]/, taxon.name )
67   end
68   new_url
69 end
