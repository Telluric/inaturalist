30 def file=( data )
31   unless data.respond_to?(:path)
32     return self.file.assign( data )
33   end
34   filename = if data.respond_to?(:original_filename)
35     data.original_filename
36   else
37     data.path.split( File::SEPARATOR ).last
38   end
39   content_type = InatContentTypeDetector.new( data.path ).detect
40   # For whatever reason, the file command tends to recognize mp3 files as
41   # audio/mpeg, mp4 files as video/mp4, and m4a files as audio/x-m4a. The
42   # intent here is to not convert wav and mp3 files
43   if [
44     "audio/wav",
45     "audio/wave",
46     "audio/x-wav",
47     "audio/mpeg"
48   ].include?( content_type )
49     self.file.assign( data )
50   else
51     file_format = File.extname( data.path )
52     file_basename = File.basename( filename, file_format )
53     dest_path = dst_path = File.join( Dir.tmpdir, "#{file_basename}.m4a" )
54     # begin
55     # -i specifies the input
56     # -vn ensures the output has no video
57     # -y overwrites the output path regardless of what's there
58     # -strict -2 allows use of an experimental AAC codec on linux
59     line = Terrapin::CommandLine.new(
60       "ffmpeg",
61       "-i :source -vn -y -strict -2 -compression_level 0 :dest"
62     )
63     line.run(
64       source: data.path,
65       dest: dest_path
66     )
67     File.open( dest_path, "rb" ) do |f|
68       self.file.assign( f )
69     end
70   end
71 end
