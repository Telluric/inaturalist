100 def merge(reject, options = {})
101   return false if reject.project_observation_fields.exists?
102   attrs_to_merge = (options[:merge] || []).map(&:to_s)
103   attrs_to_keep = (options[:keep] || []).map(&:to_s)
104   %w(name datatype allowed_values description).each do |a|
105     if attrs_to_merge.include?(a)
106       new_value = if a == 'allowed_values'
107         rejected_values = reject.allowed_values.split('|')
108         all_values = (allowed_values.split('|') + rejected_values).uniq.join('|')
109         all_values
110       else
111         [send(a).to_s, reject.send(a).to_s].join(' ')
112       end
113       self.send("#{a}=", new_value)
114     elsif attrs_to_keep.include?(a)
115       self.send("#{a}=", reject.send(a))
116     end
117   end
118   if changed
119     unless save
120       Rails.logger.debug "[DEBUG] Failed to save: #{errors.full_messages.to_sentence}"
121     end
122   end
123   reject.observation_field_values.update_all(:observation_field_id => id)
124   reject.destroy    
125 end
