169 def get_flaggable_user
170   case flaggable_type
171   when "User" then flaggable
172   when "Message" then flaggable.from_user
173   else
174     k, reflection = flaggable.class.reflections.detect{|r| r[1].class_name == "User" && r[1].macro == :belongs_to }

176       flaggable.send( k )
177     else
178       nil
179     end
180   end
181 end
