37 def content_flagged_as_spam
38   # The FlagsController is apparently the place to check for what
39   # models use the acts_as_flaggable module
40   Rakismet.spammable_models.map{ |klass|
41     # User model is a "spammable" model, but users are not content
42     next if klass == User
43     # classes have different ways of getting to user, so just do
44     # a join and enforce the user_id with a where clause
45     klass.joins(:user).where(users: { id: self.id }).
46       joins(:flags).where({ flags: { flag: Flag::SPAM, resolved: false } })
47   }.compact.flatten.uniq
48 end
