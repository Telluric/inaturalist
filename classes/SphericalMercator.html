   # File lib/spherical_mercator.rb
66 def from_ll_to_pixel(lonlat, zoom, options = {})
67   lonlat[1] = -89.999 if lonlat[1] <= -90
68   lonlat[1] = 90 if lonlat[1] > 90
69   e = @zc[zoom]
70   x = lonlat[0] * @Bc[zoom] + e[0]
71   g = Math.log(Math.tan((lonlat[1] / (2*@RAD_TO_DEG)) + (Math::PI / 4)))
72   y = e[1] - (g * @Cc[zoom])
73   if options[:skip_round]
74     [x, y]
75   else
76     [x.round, y.round]
77   end
78 end

58 def from_pixel_to_ll(px, zoom)
59   e = @zc[zoom]
60   f = (px[0] - e[0]) / @Bc[zoom]
61   g = (px[1] - e[1]) / -@Cc[zoom]
62   h = @RAD_TO_DEG * (2 * Math.atan(Math.exp(g)) - 0.5 * Math::PI)
63   return f, h
