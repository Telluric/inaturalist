 73 def by_login
 74   block_if_spammer(@selected_user) && return
 75   params[:page] = params[:page].to_i
 76   params[:page] = 1 unless params[:page] > 0
 77   user_filter = { term: { "user.id": @selected_user.id } }
 78   ownership_filter = { term: { "own_observation": false } }
 79   date_parts = Identification.conditions_for_date("col", params[:on])
 80   # only if conditions_for_date determines a valid range will it return
 81   # an array of [ condition_to_interpolate, min_date, max_date ]
 82   if date_parts.length == 3
 83     date_filters = [
 84       { range: { "created_at": { gte: date_parts[1] } } },
 85       { range: { "created_at": { lte: date_parts[2] } } }
 86     ]
 87   end
 88   result = Identification.elastic_search(
 89     filters: [ user_filter, date_filters, ownership_filter ].flatten.compact,
 90     size: limited_per_page,
 91     from: (params[:page] - 1) * limited_per_page,
 92     sort: {
 93       "created_at": {
 94         order: "desc",
 95         mode: "max"
 96       }
 97     },
 98     track_total_hits: true,
 99     source: ["id"]
100   )
101   ids = result.response.hits.hits.map do |h|
102     h._source.id
103   end
104   # fetch the Identification instances and preload
105   instances = Identification.where(id: ids).order(id: :desc).includes(
106     { observation: [ :user, :photos, { taxon: [{taxon_names: :place_taxon_names}, :photos] } ] },
107     { taxon: [{taxon_names: :place_taxon_names}, :photos] },
108     :user
109   )
110   # turn the instances into a WillPaginate Collection
111   @identifications = WillPaginate::Collection.create(params[:page], limited_per_page,
112     result.response.hits.total.value) do |pager|
113     pager.replace(instances.to_a)
114   end
115   respond_to do |format|
116     format.html do
117       @identifications_by_obs_id = @identifications.index_by(&:observation_id)

119       render layout: "bootstrap"
120     end

123       @identifications.each do |i|
124         i.taxon.current_user = current_user if i.taxon && current_user
125         i.observation.taxon.current_user = current_user if i.observation.taxon && current_user
126         i.observation.localize_place = current_user.try(:place) || @site.place
127         i.observation.localize_locale = current_user.try(:locale) || @site.locale
128       end
129       taxon_options = {
130         only: [:id, :name, :rank],
131         methods: [:default_name, :photo_url, :iconic_taxon_name]
132       }
133       render json: @identifications.as_json( include: [

47   ids = Identification.where(id: api_response.results.map{ |r| r["id"] }).
48     includes(
49       :observation,
50       :stored_preferences,
51       :user,
52       { taxon: [
53         { taxon_names: :place_taxon_names },
54         { photos: [:flags, :file_prefix,] }
55       ] }
56     ).order(id: :desc)
57   Observation.preload_for_component(ids.map(&:observation), logged_in: logged_in?)
58   @identifications = WillPaginate::Collection.create(params[:page] || 1, per_page,
59     api_response.total_results) do |pager|
60     pager.replace(ids)
61   end
62   counts_response = INatAPIService.identifications_categories(search_params)
63   @counts = Hash[counts_response.results.map{ |r| [ r["category"], r["count"] ] }]

65     format.html { render layout: "bootstrap" }
