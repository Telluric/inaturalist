 2 def job_can_use_replica( job )
 3   return true if job.handler_yaml.is_a?( ObservationsExportFlowTask )
 4   unless job.handler_yaml.respond_to?(:object) && job.handler_yaml.respond_to?(:method_name)
 5     return false
 6   end
 7   class_name = ( job.handler_yaml.object.kind_of?( ActiveRecord::Base ) ?
 8     job.handler_yaml.object.class.name : job.handler_yaml.object.name ) rescue nil
 9   object_method = "#{class_name}::#{job.handler_yaml.method_name}"
10   return true if object_method == "Identification::run_update_curator_identification"
11   return true if object_method == "Identification::update_categories_for_observation"
12   return true if object_method == "CheckList::refresh_listed_taxon"
13   return true if object_method == "CheckList::refresh"
14   return true if object_method == "Observation::notify_subscribers_of"
15   return true if object_method == "Identification::notify_subscribers_of"
16   return true if object_method == "UpdateAction::email_updates_to_user"
17   false
18 end
