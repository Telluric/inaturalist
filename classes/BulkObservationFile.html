212 def collate_errors( exception )
213   # enumerate the exceptions and collate error messages
214   field_options = {}
215   errors = {}
216   exception.errors.each do | e |
217     if e.errors.is_a?( ActiveModel::Errors )
218       # e.errors.each do |field, error|
219       e.errors.each do | error |
220         errors[error.attribute] ||= {}
221         full_error = e.errors.full_message( error.attribute, error.message )
222         errors[error.attribute][full_error] ||= []
223         errors[error.attribute][full_error] << e.row_count
224       end
225     elsif !e.tag.nil?
226       e.errors.each do | error |
227         errors[e.tag] ||= {}
228         errors[e.tag][error] ||= []
229         errors[e.tag][error] << e.row_count
230       end
231     else
232       e.errors.each do | error |
233         errors["base"] ||= {}
234         errors["base"][error] ||= []
235         errors["base"][error] << e.row_count
236       end
237     end
238   end
239 
240   {
241     reason: exception.reason,
242     errors: errors.stringify_keys.sort_by {| k, _v | k },
243     field_options: field_options
244   }
245 end

117 def import_file
118   row_count = 1
119   observations = []
120   ActiveRecord::Base.transaction do
121     CSV.foreach( @observation_file, encoding: "iso-8859-1:utf-8", headers: true ) do | row |
122       next if skip_row?( row )
123 
124       # Add the observation file name as a tag for identification purposes.
125       tags = row[6].blank? ? [] : row[6].split( "," )
126       tags << File.basename( @observation_file )
127       row[6] = tags.join( "," )
128 
129       obs = new_observation( row )
130       # Not sure why but without this the OFVs won't save as of Rails 6 ~~~kueda 20211028
131       obs.observation_field_values.to_a
132       begin
133         # Try to save the observation
134         obs.save!
135 
136         # Add this observation to a list for later importing to the project.
137         observations << obs
138 
139         # Increment the row count so we can tell them where any errors are.
140         row_count += 1
141       rescue ActiveRecord::RecordInvalid
142         raise BulkObservationException.new( "Invalid record encountered", row_count )
143       end
144     end
145 
146     # Add all of the observations to the project if a project was specified
147     if project
148       observations.each do | obs |
149         project.project_observations.create( observation: obs )
150       end
151 
152       # Manually update counter caches.
153       ProjectUser.update_observations_counter_cache_from_project_and_user( project.id, user.id )
154       ProjectUser.update_taxa_counter_cache_from_project_and_user( project.id, user.id )
155       Project.update_observed_taxa_count( project.id )
156     end
157   end
158   Observation.elastic_index!( ids: observations.map( &:id ), wait_for_index_refresh: true )
159 end

 72 def validate_file
 73   row_count = 1
 74   errors = []
 75 
 76   # Parse the entire observation file looking for possible errors.
 77   begin
 78     CSV.foreach( @observation_file, encoding: "iso-8859-1:utf-8", headers: true ) do | row |
 79       next if skip_row?( row )
 80 
 81       # Look for the species and flag it if it's not found.
 82       taxon = Taxon.single_taxon_for_name( row[0] )
 83       if taxon.nil?
 84         errors << BulkObservationException.new( "Single taxon not found: #{row[0]}", row_count + 1, [],
 85           "species_not_found" )
 86       end
 87 
 88       # Check the validity of the observation
 89       obs = new_observation( row )
 90       unless obs.valid?
 91         errors << BulkObservationException.new( "Observation is not valid", row_count + 1, obs.errors )
 92       end
 93 
 94       # Increment the row count.
 95       row_count += 1
 96 
 97       # Stop if we have reached our max error count
 98       break if errors.count >= MAX_ERROR_COUNT
 99     end
100   rescue CSV::MalformedCSVError => e
101     line = e.message[/line (\d+)/, 1]
102     errors << BulkObservationException.new( e.message, line, [e] )
103   end
104   if errors.count.positive?
105     raise BulkObservationException.new(
106       I18n.t( :we_tried_to_process_your_upload_named_filename, filename: File.basename( @observation_file ) ),
107       nil,
108       errors
109     )
110   end
111   return unless row_count.zero?
112 
113   raise BulkObservationException, "The observation file '#{File.basename( @observation_file )}' was empty."
114 end
