merge_duplicates( options = {} ) Link

This will remove all duplicates by grouping on project_id and user_id. Probably only useful when it gets used by User.merge, otherwise probably rather dangerous

207   def self.merge_duplicates( options = {} )
208     debug = options.delete(:debug)
209     where = options.map{|k,v| "#{k} = #{v}"}.join(" AND ") unless options.blank?
210     sql = <<-SQL
211       SELECT project_id, user_id, array_agg(id) AS ids, count(*)
212       FROM project_users
213       #{"WHERE #{where}" if where}
214       GROUP BY project_id, user_id HAVING count(*) > 1
215     SQL
216     puts "Finding project_users WHERE #{where}" if debug
217     connection.execute( sql.gsub(/\s+/, " " ).strip ).each do |row|
218       to_merge_ids = row["ids"].to_s.gsub( /[\{\}]/, "" ).split( "," ).sort
219       pu = ProjectUser.find_by_id( to_merge_ids.first )
220       puts "pu: #{pu}, merging #{to_merge_ids}" if debug
221       rejects = ProjectUser.where( id: to_merge_ids[1..-1] )
222       rejects.destroy_all
223     end
224   end

226   def self.merge_future_duplicates( reject, keeper )
227     Rails.logger.debug "[DEBUG] ProjectUser.merge_future_duplicates, reject: #{reject}, keeper: #{keeper}"
228     unless reject.is_a?( keeper.class )
229       raise "reject and keeper must by of the same class"
230     end
231     unless reject.is_a?( User )
232       raise "ProjectUser.merge_future_duplicates only works for observations right now"
233     end
234     k, reflection = reflections.detect{|k,r| r.klass == reject.class && r.macro == :belongs_to }
235     sql = <<-SQL
236       SELECT
237         project_id,
238         array_agg(id) AS ids
239       FROM
240         project_users
241       WHERE
242         #{reflection.foreign_key} IN (#{reject.id},#{keeper.id})
243       GROUP BY
244         project_id
245       HAVING
246         count(*) > 1
247     SQL
248     connection.execute( sql.gsub(/\s+/, " " ).strip ).each do |row|
249       to_merge_ids = row['ids'].to_s.gsub(/[\{\}]/, '').split(',').sort
250       project_users = ProjectUser.where( id: to_merge_ids )
251       if reject_pu = project_users.detect{|i| i.send(reflection.foreign_key) == reject.id }
252         reject_pu.destroy
253       end
254     end
255   end
