Create PlaceTaxonNames for matching countries. This helps people who cannot choose a locale that matches a lexicon but can choose a place.

22 def self.create_country_records_from_lexicons(options = {})
23   start = Time.now
24   logger = options[:logger] || Rails.logger
25   created = 0
26   errors = 0
27   mapping = options[:mapping] || {
28     Japanese: [:Japan],
29     German: [:Germany, :Austria],
30     "Chinese Traditional" => ['Hong Kong', :Taiwan],
31     "Chinese Simplified" => [:China]
32   }
33   mapping.each do |lexicon, country_names|
34     countries = Place.where(admin_level: Place::COUNTRY_LEVEL, name: country_names).to_a
35     TaxonName.joins("LEFT OUTER JOIN place_taxon_names ptn ON ptn.taxon_name_id = taxon_names.id").
36         includes(:place_taxon_names).
37         where("taxon_names.lexicon = ?", lexicon).find_each do |tn|
38       # not a fan of the overselection and filter approach here, since we have a lot of names. Is there a way to do this in the db?
39       candidate_countries = countries.select{|c| !tn.place_ids.include?(c.id)}
40       next if candidate_countries.blank?
41       candidate_countries.each do |country|
42         ptn = PlaceTaxonName.new(taxon_name: tn, place: country)
43         if ptn.save
44           logger.info "Added #{tn} to #{country}"
45           created += 1
46         else
47           logger.error "[ERROR] Failed to save #{ptn}: #{ptn.errors.full_messages.to_sentence}"
48           errors += 1
49         end
50       end
51     end
52   end
53   logger.info "Created #{created} PlaceTaxonName records, failed on #{errors} (#{Time.now - start}s)"
54 end
