 48 def create
 49   auth_info = request.env['omniauth.auth']
 50   
 51   case auth_info["provider"]
 52   when 'facebook'
 53     # omniauth bug sets fb nickname to 'profile.php?id=xxxxx' if no other nickname exists
 54     if auth_info["info"]["nickname"] && auth_info["info"]["nickname"].match("profile.php")
 55       auth_info["info"]["nickname"] = nil
 56     end
 57     # for some reason, omniauth doesn't populate image url
 58     # (maybe cause our version of omniauth was pre- fb graph api?)
 59     auth_info["info"]["image"] = "http://graph.facebook.com/#{auth_info["uid"]}/picture?type=large"
 60   when 'flickr'
 61     if auth_info["info"]["image"].blank?
 62       # construct the url for the user's flickr buddy icon
 63       nsid = auth_info['uid']
 64       flickr_info = flickr.people.getInfo(:user_id => nsid) # we make this api call to get the icon-farm and icon-server
 65       iconfarm = flickr_info['iconfarm']
 66       iconserver = flickr_info['iconserver']
 67       unless (iconfarm == 0 || iconserver == 0)
 68         auth_info["info"]["image"] = "http://farm#{iconfarm}.static.flickr.com/#{iconserver}/buddyicons/#{nsid}.jpg"
 69       end
 70     end
 71   when "apple"
 72     if auth_info["info"]["name"] && auth_info["info"]["name"]["firstName"]
 73       full_name = auth_info["info"]["name"]["firstName"]
 74       if auth_info["info"]["name"]["lastName"]
 75         full_name += " " + auth_info["info"]["name"]["lastName"]
 76       end
 77       auth_info["info"]["name"] = full_name
 78     end
 79   end
 80   
 81   
 82   if @provider_authorization = ProviderAuthorization.find_from_omniauth(auth_info)
 83     update_existing_provider_authorization(auth_info)
 84   else
 85     create_provider_authorization(auth_info)
 86   end
 87   
 88   if @provider_authorization && @provider_authorization.valid? && (scope = get_session_omniauth_scope)
 89     @provider_authorization.update(:scope => scope.to_s)
 90     session["omniauth_#{request.env['omniauth.strategy'].name}_scope"] = nil
 91   end
 92   
 93   if !session[:return_to].blank? && session[:return_to] != login_url
 94     @landing_path ||= session[:return_to]
 95   end
 96   
 97   # registering via an invite link in a flickr comment. see /flickr/invite
 98   if session[:invite_params]
 99     invite_params = session[:invite_params]
100     session[:invite_params] = nil
101     if @provider_authorization && @provider_authorization.valid? && @provider_authorization.created_at > 15.minutes.ago
102       flash[:notice] = "Welcome to #{@site.name}! If these options look good, " +
103         "click \"Save observation\" below and you'll be good to go!"
104       invite_params.merge!(:welcome => true)
105     end
106     @landing_path = new_observation_url(invite_params)
107   end
108   redirect_to @landing_path || home_url
109 end

20 def destroy
21   return render_404 unless request.delete?
22   provider_authorization = current_user.has_provider_auth(params[:provider]) unless params[:provider].blank?
23   provider_authorization ||= current_user.provider_authorizations.find_by_id( params[:id] )
24   provider_authorization.destroy if provider_authorization
25   respond_to do |format|
26     format.html do
27       flash[:notice] = if provider_authorization
28         t(:you_unlinked_your_provider_account,
29           provider: ProviderAuthorization::PROVIDER_NAMES[provider_authorization.provider_name]
30         )
31       else
32         t(:failed_to_unlink_your_account)
33       end
34       redirect_to edit_person_url( current_user )
35     end
36     format.json do
37       if provider_authorization
38         head :no_content
39       else
40         render status: :unprocessable_entity, json: {
41           error: t(:failed_to_unlink_account),
42         }
43       end
44     end
45   end
46 end
