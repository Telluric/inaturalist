294   end
295 
296   unless page
297     search_results = eol.search(name, :exact => true)
298     if result = search_results.search( "result" ).first
299       page = eol.page( result.at( "id" ).content, page_request_params )
300     end
301   end
302

221   def sync_eol_sections(options = {})
222     return unless page = get_eol_page(options)
223     subjects = (options[:subjects] || []).select{|s| !s.blank?}
224 
225     xpath = <<-XPATH
226       //dataObject[
227         descendant::dataType[text()='http://purl.org/dc/dcmitype/Text']
228         and descendant::subject
229       ]
230     XPATH
231     locale = (options[:locale] || guide.user.locale || I18n.locale).to_s
232     data_objects = page.xpath(xpath).reject do |data_object|
233       data_object.at('language') && locale !~ /^#{data_object.at('language').content}/
234     end
235 
236     guide_sections.destroy_all if options[:replace].yesish?
237     max_pos = guide_sections.calculate(:maximum, :position) || 0
238 
239     data_objects = if subjects.blank? || options[:overview] == "true"
240       [data_objects.first]
241     else
242       unique_data_objects = ActiveSupport::OrderedHash.new
243       data_objects.each do |data_object| 
244         # data_object_subject = if (s = data_object.at('subject'))
245         #   s.content.split('#').last
246         # end
247         data_object_subjects = data_object.search( "subject/subject" ).map{|s| s.content}
248         # if subjects.include?(data_object_subject)
249         if ( data_object_subjects & subjects ).size > 0
250           unique_data_objects[data_object_subject] ||= data_object
251         end
252       end
253       unique_data_objects.values
254     end
255 
256     data_objects.compact.each do |data_object|
257       gs = GuideSection.new_from_eol_data_object(data_object)
258       gs.position = (max_pos += 1)
259       if gs && !guide_sections.where(:source_url => gs.source_url).exists?
260         gs.modified_on_create = false
261         self.guide_sections << gs
262       end
263     end
264   end
