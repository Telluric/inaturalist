 50 def self.new_from_api_response(api_response, options = {})
 51   api_response.remove_namespaces! if api_response.respond_to?(:remove_namespaces!)
 52   native_photo_id = api_response.at( "dataObjectVersionID" ).try(:content)
 53   return unless native_photo_id
 54   if license = api_response.at( "license" )
 55     license_string = license.content
 56     license_number = Photo::C
 57     if license_string.include? "publicdomain"
 58       license_number = PD
 59     end
 60     [1,2,3,4,5,6].each do |num|
 61       if license_string.include? "licenses/#{Photo.license_code_for_number(num).split("CC-")[1].downcase}/"
 62         license_number = num 
 63       end
 64     end
 65   else
 66     # all EOL content should be CC licensed or PD, so if there's not license info we can assume it's PD
 67     license_number = PD
 68   end
 69   square_url, thumb_url, small_url, medium_url, original_url = nil
 70   image_url = api_response.at( "eolMediaURL" ).content

 74     medium_url = image_url.gsub( /(\.\w+?)$/, ".580x360\\1" )
 75   end

 77   rights_holder = api_response.search( "dataObject rightsHolder" )

 79     agent = api_response.search('agent[role=creator]').first
 80     agent ||= api_response.search('agent[role=photographer]').first
 81     native_username = agent.inner_text if agent
 82   elsif rights_holder.children.size > 0

14 def self.search_eol(query, options = {})
15   return [] if query.blank?
16   eol_page_id = options[:eol_page_id]
17   if eol_page_id.blank?
18     search_xml = eol.search(query, :exact => 1)
19     return [] if search_xml.blank?
20     eol_page_id = search_xml.at("entry/id").try(:text)
21   end
22   return [] if eol_page_id.blank?
23   per_page = (options[:per_page] || 36).to_i
24   per_page = 100 if per_page > 100
25   eol_page_xml = begin
26     eol.page(eol_page_id,
27       licenses: "all",
28       images_page: options[:page],
29       images_per_page: per_page,
30       maps_per_page: 0,
31       texts_per_page: 0,
32       videos_per_page: 0,
33       sounds_per_page: 0,
34       details: 1
35     )
36   rescue OpenURI::HTTPError => e
37     return []
38   end
39   eol_page_xml.xpath('//xmlns:dataObject[.//xmlns:mediaURL]').map do |data_object|
40     new_from_api_response(data_object)
41   end.compact
42 end

110 def repair(options = {})
