283 def add_taxon_batch
284   return redirect_to(@list) unless params[:names]
285   @added_taxa = []
286   @lines = params[:names].split("\n").to_a.map{|n| n.strip.gsub(/\s+/, " ")}.delete_if(&:blank?)
287   @max = 1000
288   @batch_size = 50
289   
290   if @lines.size > @max
291     flash[:notice] = t(:sorry_you_can_only_add_1000_at_a_time)
292     return redirect_to(@list)
293   end
294   
295   @errors = []
296   @lines_taxa = []
297   @lines.each_with_index do |name, i|
298     
299     # If we're past 50, just add nil and assume we'll deal with it later
300     if i > @batch_size
301       @lines_taxa << [name, nil]
302       next
303     end
304     
305     taxon_names = TaxonName.joins(:taxon).includes(:taxon).
306       where("taxa.id IS NOT NULL AND lower(taxon_names.name) = ?", name.to_s.downcase).
307       page(1)
308     case taxon_names.size
309     when 0
310       @lines_taxa << [name, "not found"]
311     when 1
312       listed_taxon = @list.add_taxon(taxon_names.first.taxon, :user_id => current_user.id, :manually_added => true)
313       if listed_taxon.valid?
314         @lines_taxa << [name, listed_taxon]
315       else
316         @lines_taxa << [name, listed_taxon.errors.full_messages.to_sentence]
317       end
318     else
319       @lines_taxa << [name, "matched several different taxa"]
320     end
321   end
322   
323   respond_to do |format|
324     format.html { render :template => "lists/add_taxon_batch" }
325     format.json do
326       json = @lines_taxa.map do |name, item|
327         {
328           :name => name,
329           :parameterized_name => name.parameterize,
330           :item => item, 
331           :html => view_context.render_in_format(:html, :partial => "lists/add_taxon_batch_line", :object => [name, item])
332         }
333       end
334       render :json => json
335     end
336   end
