19     def load_geojson_file( geojson_path, options = {} )
20       logger = options[:logger] || Rails.logger
21       return unless File.exist?( geojson_path )
22       pg_string = {
23         dbname: ApplicationRecord.connection_db_config.configuration_hash[:database],
24         host: ApplicationRecord.connection_db_config.configuration_hash[:host],
25         user: ApplicationRecord.connection_db_config.configuration_hash[:username],
26         password: ApplicationRecord.connection_db_config.configuration_hash[:password],
27       }.map { |k, v| "#{k}=#{v}" }.join( " " )
28       # Note that ogr2ogr will automatically create a spatial index on the geom column
29       cmd = <<-BASH
30         ogr2ogr -f "PostgreSQL" PG:"#{pg_string}" \
31           #{geojson_path} \
32           -nln #{table_name} \
33           -nlt MULTIPOLYGON \
34           -lco GEOMETRY_NAME=geom \
35           -overwrite
36       BASH
37       logger.info "Loading time zones..."
38       if system( cmd ) && TimeZoneGeometry.count > 0
39         logger.info "Loaded #{TimeZoneGeometry.count} time zones with ogr2ogr"
40       else
41         logger.info "ogr2ogr failed for some reason, try to load and process with RGeo instead"
42         File.open( geojson_path ) do |f|
43           logger.info "Reading GeoJSON from #{geojson_path}..."
44           json = RGeo::GeoJSON.decode( f.read )
45           TimeZoneGeometry.transaction do
46             logger.info "Truncating #{table_name}..."
47             TimeZoneGeometry.connection.execute( "TRUNCATE TABLE #{table_name} RESTART IDENTITY" )
48             logger.info "Loading #{json.size} time zones"
49             json.each do |zone|
50               print "." if options[:debug]
51               geom = if zone.geometry.geometry_type == ::RGeo::Feature::Polygon
52                 factory = RGeo::Cartesian.simple_factory( srid: 0 )
53                 factory.multi_polygon( [zone.geometry] )
54               else
55                 zone.geometry
56               end
57               TimeZoneGeometry.create!( tzid: zone.properties["tzid"], geom: geom )
58             end
59             logger.info
60           end
61         end
62         logger.info "Loaded #{TimeZoneGeometry.count} time zones with RGeo"
63       end
64 
65       # By default the SRID is 4326, and all ours are 0 for some reason
66       logger.info "Resetting SRID..."
67       ActiveRecord::Base.connection.execute "SELECT UpdateGeometrySRID('#{table_name}', 'geom', 0)"
68     end
