36 def self.new_from_api_response(api_response, options = {})
37   base = "http://maps.googleapis.com/maps/api/streetview?"
38   params = api_response.merge(:sensor => false)
39   w,h = (api_response[:size] || "1x1").split('x').map(&:to_f)
40   r = h / w rescue 1
41   r = 1 if r > 1 || r <= 0
42   z = (2 - (2 * (params[:fov] || 90).to_f / 180.0)).floor
43   options.update(
44     :thumb_url => "#{base}#{params.merge(:size => "75x#{(75*r).to_i}").map{|k,v| "#{k}=#{v}"}.join('&')}",
45     :square_url => "#{base}#{params.merge(:size => "100x100").map{|k,v| "#{k}=#{v}"}.join('&')}",
46     :small_url => "#{base}#{params.merge(:size => "240x#{(240*r).to_i}").map{|k,v| "#{k}=#{v}"}.join('&')}",
47     :medium_url => "#{base}#{params.merge(:size => "500x#{(500*r).to_i}").map{|k,v| "#{k}=#{v}"}.join('&')}",
48     :large_url => "#{base}#{params.merge(:size => "640x#{(640*r).to_i}").map{|k,v| "#{k}=#{v}"}.join('&')}",
49     :original_url => "#{base}#{params.map{|k,v| "#{k}=#{v}"}.join('&')}",
50     :native_realname => "Google",
51     :native_page_url => "https://maps.google.com/?ll=#{params[:location]}&layer=c&cbll=#{params[:location]}&cbp=12,#{params[:heading]},,#{z},#{params[:pitch].to_f * -1}",
52     :license => Photo::COPYRIGHT
53   )
54   options[:native_photo_id] ||= options[:original_url]
55   GoogleStreetViewPhoto.new(options)
56 end
