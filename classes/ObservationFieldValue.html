319 def self.update_for_taxon_change( taxon_change, options, &block )
320   input_taxon_ids = taxon_change.input_taxa.map(&:id)
321   scope = ObservationFieldValue.joins( :observation_field ).
322     where( "observation_fields.datatype = ?", ObservationField::TAXON )
323   scope = scope.where(
324     input_taxon_ids.map {|itid| "observation_field_values.value = '#{itid}'" }.join( " OR " )
325   )
326   scope = scope.where( user_id: options[:user] ) if options[:user]
327   scope = scope.where( "observation_field_values.id IN (?)", options[:records] ) unless options[:records].blank?
328   scope = scope.where( options[:conditions] ) if options[:conditions]
329   scope = scope.includes( options[:include] ) if options[:include]
330   obs_ids = Set.new
331   scope.find_each do |ofv|
332     next unless output_taxon = taxon_change.output_taxon_for_record( ofv )
333     next if !taxon_change.automatable_for_output?( output_taxon.id )
334     ofv.update( value: output_taxon.id )
335     obs_ids << ofv.observation_id
336   end
337   Observation.elastic_index!( ids: obs_ids.to_a )
338 end

106 def validate_observation_field_datatype
107   return true if observation.mobile? # HACK until we implement ui for *all* observation field types in the mobile apps
108   case observation_field.datatype
109   when "numeric"
110     errors.add(:value, "must be a number") if value !~ Observation::FLOAT_REGEX
111   when "location"
112     errors.add(:value, "must decimal latitude and decimal longitude separated by a comma") if value !~ LAT_LON_REGEX
113   when "date"
114     unless value =~ /^\d{4}\-\d{2}\-\d{2}$/
115       errors.add(:value, :date_format, 
116         :observation_field_name => observation_field.name,
117         :observation_species_guess => observation.try(:species_guess)
118       )
119     end
120   when "datetime"
121     begin
122       Time.iso8601(value)
123     rescue ArgumentError => e
124       errors.add(:value, :datetime_format,
125         :observation_field_name => observation_field.name,
126         :observation_species_guess => observation.try(:species_guess)
127       )
128     end
129   when "time"
130     begin
131       Time.parse(value)
132       if value !~ /^\d\d:\d\d/
133         errors.add(:value, :time_format, 
134           :observation_field_name => observation_field.name,
135           :observation_species_guess => observation.try(:species_guess)
136         )
137       end
138     rescue ArgumentError => e
139       errors.add(:value, :time_format, 
140         :observation_field_name => observation_field.name,
141         :observation_species_guess => observation.try(:species_guess)
142       )
143     end
144   when "dna"
145     if value =~ /[^ATCG\s]/
146       errors.add(:value, :dna_only_atcg)
147     end
148   end
