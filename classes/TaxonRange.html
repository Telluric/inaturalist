 99 def create_geom_from_kml_attachment
100   return unless File.exists?(self.range.path)
101   tmp_path = File.join(Dir::tmpdir, "#{self.id}_#{Time::now.seconds_since_midnight.round}.geojson")
102   cmd = "ogr2ogr -f GeoJSON #{tmp_path} #{self.range.path}"
103   system cmd
104   open(tmp_path) do |f|
105     if geojsongeom = GeoRuby::SimpleFeatures::Geometry.from_geojson(f.read)
106       self.geom = geojsongeom.features.first.geometry.as_wkt
107       if geom && geom.geometry_type == RGeo::Feature::Polygon
108         factory = RGeo::Cartesian.simple_factory( srid: 0 )
109         self.geom = factory.multi_polygon([geom])
110       end
111       self.geom_processed = true
112       self.save
113     end
114     f.close
115   end
116   File.delete(tmp_path)
117 end

75 def create_kml_attachment
76   return unless geom
77   wkt = RGeo::WKRep::WKTGenerator.new(convert_case: :upper).generate(geom)
78   builder = Nokogiri::XML::Builder.new do |xml|
79     xml.kml('xmlns' => 'http://earth.google.com/kml/2.1') do
80       xml.Document {
81         xml.Placemark {
82           xml.name
83           xml.description
84           xml.styleUrl "#{Site.default.url}/assets/index.kml#taxon_range"
85           xml << GeoRuby::SimpleFeatures::Geometry.from_ewkt(wkt).as_kml
86         }
87       }
88     end
89   end
90   tmp_path = File.join(Dir::tmpdir, "temp.kml")
91   f = File.open(tmp_path, "w")
92   f.write(builder.to_xml)
93   f.close
94   file = File.open(tmp_path, "r")
