 6 def self.fetch(flickraw, type, method, params={})
 7   # find or create the endpoint
 8   @api_endpoint = ApiEndpoint.find_or_create_by(title: "Flickr",
 9     documentation_url: "http://www.flickr.com/services/api/",
10     base_url: "https://api.flickr.com/services/rest?", cache_hours: 720)
11   # find or create the cache entry for this call
12   api_endpoint_cache = ApiEndpointCache.find_or_create_by(api_endpoint: @api_endpoint,
13     request_url: "flickr.#{ type }.#{ method }(#{ params.to_json })")
14   # return the cached entry if it is valid
15   api_response = if api_endpoint_cache.cached?
16     # it may have cached nil if there was a problem
17     if cached_response = api_endpoint_cache.response
18       # JSON.parse(api_endpoint_cache.response).map do |r|
19       #   # map each hash to an object, as we would get from Flickraw
20       #   OpenStruct.new(r)
21       # end
22       json_to_flickraw( api_endpoint_cache.response )
23     end
24   else
25     response = nil
26     begin
27       # mark the start time of the fetch
28       api_endpoint_cache.update(request_began_at: Time.now,
29         request_completed_at: nil, success: nil, response: nil)
30       # call the API via Flickraw
31       Timeout::timeout(TIMEOUT) do
32         response = request( flickraw, type, method, params )
33       end
34       # mark the end time and update the cache with the results
35       api_endpoint_cache.update(
36         request_completed_at: Time.now, success: true, response: response)
37     rescue Timeout::Error
38       raise Timeout::Error, "Flickr didn't respond within #{TIMEOUT} seconds."
39     rescue FlickRaw::FailedResponse, EOFError, OpenSSL::SSL::SSLError => e
40       Rails.logger.error "Failed Flickr API request: #{e}"
41       Rails.logger.error e.backtrace.join("\n\t")
42       api_endpoint_cache.update(request_completed_at: Time.now, success: false)
43     end
44     response ? json_to_flickraw( response ) : nil
45   end
46 end
