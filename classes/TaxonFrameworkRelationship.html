 94 def set_relationship
 95   external_taxa_count = external_taxa.count
 96   taxa_count = taxa.count
 97   if external_taxa_count == 1 && taxa_count == 1
 98     if external_taxa.first.name == taxa.first.name && 
 99        external_taxa.first.rank == taxa.first.rank &&
100        external_taxa.first.parent_name == taxa.first.parent.name &&
101        external_taxa.first.parent_rank.downcase == taxa.first.parent.rank
102       self.relationship = "match"
103     elsif external_taxa.first.name == taxa.first.name && 
104        external_taxa.first.rank == taxa.first.rank
105       self.relationship = "alternate_position"
106     else
107       self.relationship = "one_to_one"
108     end
109   elsif external_taxa_count > 1 && taxa_count > 1
110     self.relationship = "many_to_many"
111   elsif external_taxa_count == 1 && taxa_count > 1
112     self.relationship = "many_to_one"
113   elsif external_taxa_count > 1 && taxa_count == 1
114     self.relationship = "one_to_many"
115   elsif external_taxa_count == 0 && taxa_count >= 1
116     self.relationship = "not_external"
117   elsif external_taxa_count >= 1 && taxa_count == 0
118     self.relationship = "not_internal"
119   else
120     self.relationship = "error"
121   end
122   true
123 end
