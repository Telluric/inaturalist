 61 def compare
 62   @with = List.find_by_id(params[:with])
 63   @iconic_taxa = Taxon::ICONIC_TAXA
 64   
 65   unless @with
 66     flash[:notice] = t(:you_cant_compare_a_list_with_nothing)
 67     return redirect_to(list_path(@list))
 68   end
 69   
 70   find_options = {
 71     :include => [:taxon],
 72     :order => "taxa.ancestry"
 73   }
 74   
 75   # Load listed taxa for pagination
 76   paginating_find_options = find_options.merge(
 77     :page => params[:page], 
 78     :per_page => 26)
 79     
 80   left_condition = ["list_id = ?", @list]
 81   right_condition = ["list_id = ?", @with]
 82   both_condition = ["list_id in (?)", [@list, @with].compact]
 83 
 84   # TODO: pull out check list logic
 85   if @list.is_a?(CheckList) && @list.is_default?
 86     left_condition = ["place_id = ?", @list.place_id]
 87     both_condition = ["(place_id = ? OR list_id = ?)", @list.place_id, @with]
 88   end
 89   
 90   @show_list = params[:show] if %w(left right).include?(params[:show])
 91   list_conditions = case @show_list
 92   when "left"   then  left_condition
 93   when "right"  then  right_condition
 94   else                both_condition
 95   end
 96   
 97   paginating_find_options[:conditions] = list_conditions
 98   
 99   # Handle iconic taxon filtering
100   if params[:taxon]
101     @filter_taxon = Taxon.find_by_id(params[:taxon])
102     paginating_find_options[:conditions] = update_conditions(paginating_find_options[:conditions], 
103       ["AND taxa.iconic_taxon_id = ?", @filter_taxon])
104   end
105   
106   @paginating_listed_taxa = ListedTaxon.where(paginating_find_options[:conditions]).
107     includes(paginating_find_options[:include]).
108     paginate(page: paginating_find_options[:page], per_page: paginating_find_options[:per_page]).
109     order(paginating_find_options[:order])
110   
111   # Load the listed taxa for display.  The reason we do diplay and paginating
112   # listed taxa is that strictly paginating and ordering by lft would sometimes
113   # leave some taxa off the end.  Say you grabbed 26 listed_taxa and the last in
114   # list1 was a Snowy Egret, but since list2 actually had more taxa than list1
115   # on this page, the page didn't include list2's snowy egret, b/c it would be
116   # on the next page.  I know, I'm confused even writing it.  KMU 2009-02-08
117   find_options[:conditions] = update_conditions(both_condition, 
118     ["AND listed_taxa.taxon_id IN (?)", @paginating_listed_taxa.map(&:taxon_id)])
119   find_options[:include] = [
120     :last_observation,
121     {:taxon => [:iconic_taxon, :photos, :taxon_names]}
122   ]
123   @listed_taxa = ListedTaxon.where(find_options[:conditions]).
124     includes(find_options[:include]).
125     order(find_options[:order])
126   
127   # Group listed_taxa into owner / other pairs
128   @pairs = []
129   @listed_taxa.group_by(&:taxon_id).each do |taxon_id, listed_taxa|
130     listed_taxa << nil if listed_taxa.size == 1
131     if @list.is_a?(CheckList) && @list.is_default?
132       listed_taxa.reverse! if listed_taxa.first.place_id != @list.place_id
133     else
134       listed_taxa.reverse! if listed_taxa.first.list_id != @list.id
135     end
136     @pairs << listed_taxa
137   end
138   
139   # Calculate the stats
140   # TODO: pull out check list logic
141   @total_listed_taxa = if @list.is_a?(CheckList) && @list.is_default?
142     ListedTaxon.distinct.where(place_id: @list.place_id).count(:taxon_id)
143   else
144     @list.listed_taxa.count
145   end
146   @with_total_listed_taxa = @with.listed_taxa.count
147   @max_total_listed_taxa = [@total_listed_taxa, @with_total_listed_taxa].max
148   @iconic_taxon_counts = get_iconic_taxon_counts(@list, @iconic_taxa)
149   @with_iconic_taxon_counts = get_iconic_taxon_counts(@with, @iconic_taxa)
150   
151   @iconic_taxon_counts_hash = {}
152   @iconic_taxon_counts.each do |iconic_taxon, count|
153     if iconic_taxon.nil?
154       @iconic_taxon_counts_hash['Undefined'] = count
155     else
156       @iconic_taxon_counts_hash[iconic_taxon] = count
157     end
158   end
159   
160   @with_iconic_taxon_counts_hash = {}
161   @with_iconic_taxon_counts.each do |iconic_taxon, count|
162     if iconic_taxon.nil?
163       @with_iconic_taxon_counts_hash['Undefined'] = count
164     else
165       @with_iconic_taxon_counts_hash[iconic_taxon] = count
166     end
167   end
168   
169   @paired_iconic_taxon_counts = @iconic_taxa.map do |iconic_taxon|
170     pair = [
171       @iconic_taxon_counts_hash[iconic_taxon], 
172       @with_iconic_taxon_counts_hash[iconic_taxon]
173     ]
174     pair.compact.empty? ? nil : [iconic_taxon, pair]
175   end.compact

179 def remove_taxon    
180   respond_to do |format|
181     if @listed_taxon = @list.listed_taxa.find_by_taxon_id(params[:taxon_id].to_i)
182       @listed_taxon.destroy
183       format.html do
184         flash[:notice] = t(:taxon_removed_from_list)

197       end

201       end
202       format.json do
203         render :status => :unprocessable_entity, :json => {:error => "That taxon isn't in this list."}
