159   def self.merge_duplicates( options = {} )
160     start = Time.now
161     debug = options.delete(:debug)
162     dry = options.delete(:dry)
163     klass = self
164     where = options.map {| k,v | "#{k} = #{v}" }.join( " AND " ) unless options.blank?
165     sql = <<-SQL
166       SELECT taxon_id, place_id, authority, array_agg(id) AS ids, count(*)
167       FROM #{klass.table_name}
168       #{"WHERE #{where}" if where}
169       GROUP BY taxon_id, place_id, authority HAVING count(*) > 1
170     SQL
171     puts "Finding #{klass.name.pluralize} WHERE #{where}" if debug
172     ordered_geoprivacies = ["private", "obscured", "open", nil]
173     rejects = []
174     keepers = []
175     connection.execute( sql.gsub(/\s+/, " " ).strip ).each do |row|
176       to_merge_ids = row["ids"].to_s.gsub( /[\{\}]/, "" ).split( "," ).sort
177       records = klass.where( id: to_merge_ids ).sort_by do | a, b |
178         [ordered_geoprivacies.index(a&.geoprivacy), a&.id] <=> [ordered_geoprivacies.index(b&.geoprivacy), b&.id]
179       end.compact
180       keeper = records.shift
181       puts "keeper: #{keeper}, merging #{records}" if debug
182       keepers << keeper.id
183       rejects += records.map( &:id )
184       records.each( &:destroy ) unless dry
185     end
186     puts "#{keepers.size} kept, #{rejects.size} deleted in #{Time.now - start}" if debug
187   end
