30 def set_observation_quality_grade
31   return true unless observation
32   return true if observation.bulk_delete
33 
34   # presumably this will happen when the obs is saved
35   return true if observation.new_record?
36 
37   # If the observation will be saved after this record is created, setting
38   # quality grade should also not be necessary
39   return true if observation.will_be_saved_with_photos
40 
41   # For some reason the observation's after_commit callbacks seem to fire
42   # after the ObservationPhoto is saved, so if you don't set the quality_grade
43   # on this instance of the observation, it will fail to index properly
44   observation.quality_grade = Observation.set_quality_grade( observation.id )
45   true
