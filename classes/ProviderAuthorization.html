160 def refresh_access_token!
161   return unless provider_name == 'google_oauth2' # TODO implement for other providers
162   if refresh_token.blank?
163     Rails.logger.error "[ERROR #{Time.now}] Refresh token missing for #{self}"
164     return
165   end
166   options = {
167    body: {
168      client_id: CONFIG.google.client_id,
169      client_secret: CONFIG.google.secret,
170      refresh_token: refresh_token,
171      grant_type: 'refresh_token'
172    },
173    headers: {
174      'Content-Type' => 'application/x-www-form-urlencoded'
175    }
176   }
177   response = HTTParty.post('https://accounts.google.com/o/oauth2/token', options)
178   if response.code == 200
179     update_attribute(:token, response.parsed_response['access_token'])
180   else
181    Rails.logger.debug "[DEBUG] Failed to refresh access token, #{response.body}"
182    raise "Failed to refresh access token"
183   end
184 end

38 def uniqueness_of_authorization_per_user
39   existing_scope = if provider_uid =~ /google.com\/accounts/
40     ProviderAuthorization.
41       where(:provider_name => 'openid').
42       where("provider_uid LIKE 'https://www.google.com/accounts%'")

53   end
