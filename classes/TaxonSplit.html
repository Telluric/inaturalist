137 def output_taxon_for_record( record )
138   # note that record_place_ids includes *all* places that contain the record,
139   # so for a point that would be any place with polygons containing the point,
140   # and for places it means all ancestors of the place
141   record_place_ids = if record.respond_to?( :place ) && record.place
142     record.place.self_and_ancestor_ids
143   elsif record.is_a?( Observation ) || record.respond_to?( :observation )
144     if o = record.is_a?( Observation ) ? record : record.observation
145       o.observations_places.map(&:place_id)
146     end
147   end
148   if record_place_ids.blank?
149     return output_ancestor
150   end
151   candidate_taxa = []
152   output_taxa.each do |candidate_taxon|
153     # don't even bother looking at more taxa if you already have ambiguity
154     next if candidate_taxa.size > 1
155     # a taxon without an atlas is a candidate since we don't know where it
156     # exists
157     unless candidate_taxon.atlased?
158       candidate_taxa << candidate_taxon
159       next
160     end
161     atlas_presence_place_containing_record = candidate_taxon.cached_atlas_presence_places.detect do |p|
162       record_place_ids.include?( p.id )
163     end
164     if atlas_presence_place_containing_record
165       candidate_taxa << candidate_taxon
166     end
167   end
168   if candidate_taxa.size == 1
169     # If there's only one left standing, that's the unambiguous output taxon
170     candidate_taxa.first
171   else
172     # Otherwise, try the common ancestor of the outputs
173     output_ancestor
174   end
175 end
