elastic_index!(options = { }) Link

standard way to bulk index instances. Called without options it will page through all instances 1000 at a time (default for find_in_batches) You can also send options, including scope:

Place.elastic_index!(batch_size: 20)
Place.elastic_index!(scope: Place.where(id: [1,2,3,...]), batch_size: 20)

 62 def elastic_index!(options = { })
 63   options[:batch_size] ||=
 64     defined?(self::DEFAULT_ES_BATCH_SIZE) ? self::DEFAULT_ES_BATCH_SIZE : 1000
 65   options[:sleep] ||=
 66     defined?(self::DEFAULT_ES_BATCH_SLEEP) ? self::DEFAULT_ES_BATCH_SLEEP : 1
 67   debug = options.delete(:debug)
 68   filter_scope = options.delete(:scope)
 69   # this method will accept an existing scope
 70   scope = (filter_scope && filter_scope.is_a?(ActiveRecord::Relation)) ?
 71     filter_scope : self.all
 72   # it also accepts an array of IDs to filter by
 73   if filter_ids = options.delete(:ids)
 74     filter_ids = filter_ids.compact.uniq
 75     batch_sleep = options.delete(:sleep)
 76     if filter_ids.length > options[:batch_size]
 77       # call again for each batch, then return
 78       filter_ids.each_slice(options[:batch_size]) do |slice|
 79         elastic_index!(options.merge(ids: slice))
 80         if batch_sleep && !options[:delay]
 81           # sleep after index an ID batch, since during indexing
 82           # we only sleep when indexing multiple batches, and here
 83           # we explicitly requested a single batch to be indexed
 84           sleep batch_sleep.to_i
 85         end
 86       end
 87       return
 88     end
 89     scope = scope.where(id: filter_ids)
 90   end
 91   if indexed_before = options.delete(:indexed_before)
 92     if column_names.include?( "last_indexed_at" )
 93       scope = scope.where("last_indexed_at IS NULL OR last_indexed_at < ?", indexed_before)
 94     end
 95   end
 96   # indexing can be delayed
 97   if options.delete(:delay)
 98     # make sure to fetch the results of the scope and store
 99     # the resulting IDs instead of scopes for DelayedJobs.
100     # For example, delayed calls this like are very efficient:
101     #   Observation.elastic_index!(scope: User.find(1).observations, delay: true)
102     result_ids = scope.order(:id).pluck(:id)
103     return unless result_ids.any?
104     id_hash = Digest::MD5.hexdigest( result_ids.join( "," ) )
105     queue = if result_ids.size > 50
106       "throttled"
107     end
108     return self.delay(
109       unique_hash: { "#{self.name}::delayed_index": id_hash },
110       queue: queue
111     ).elastic_index!( options.merge(
112       ids: result_ids,
113       indexed_before: 5.minutes.from_now.strftime("%FT%T"),
114       wait_for_index_refresh: true
115     ) )
116   end
117   # now we can preload all associations needed for efficient indexing
118   if self.respond_to?(:load_for_index)
119     scope = scope.load_for_index
120   end
121   wait_for_index_refresh = options.delete(:wait_for_index_refresh)
122   batch_sleep = options.delete(:sleep)
123   batches_indexed = 0
124   scope.find_in_batches(options) do |batch|
125     if batch_sleep && batches_indexed > 0
126       # sleep only if more than one batch is being indexed
127       sleep batch_sleep.to_i
128     end
129     if debug && batch && batch.length > 0
130       Rails.logger.info "[INFO #{Time.now}] Starting to index #{self.name} :: #{batch[0].id}"
131     end
132     bulk_index(batch, wait_for_index_refresh: wait_for_index_refresh)
133     batches_indexed += 1
134   end
135   __elasticsearch__.refresh_index! if Rails.env.test?
136 end

174 def elastic_prune(start_id, end_id, options)
175   return if !start_id || !end_id || start_id >= end_id
176   options[:batch_size] ||=
177     defined?(self::DEFAULT_ES_BATCH_SIZE) ? self::DEFAULT_ES_BATCH_SIZE : 1000
178   batch_start_id = start_id
179   while batch_start_id <= end_id
180     Rails.logger.debug "[DEBUG] Processing from #{ batch_start_id }"
181     batch_end_id = batch_start_id + options[:batch_size]
182     if batch_end_id > end_id + 1
183       batch_end_id = end_id + 1
184     end
185     scope = self.where("id >= ? AND id < ?", batch_start_id, batch_end_id)
186     batch_ids = scope.pluck(:id)
187     ids_only_in_es = elastic_search(
188       sort: { id: :asc },
189       size: options[:batch_size],
190       filters: [
191         { range: { id: { gte: batch_start_id } } },
192         { range: { id: { lt: batch_end_id } } },
193         { bool: {
194           must_not: {
195             terms: { id: batch_ids }
196           }
197         } }
198       ],
199       source: ["id"]
200     ).map(&:id)
201     unless ids_only_in_es.empty?
202       Rails.logger.debug "[DEBUG] Deleting vestigial docs in ES: #{ ids_only_in_es }"
203       elastic_delete_by_ids!( [ids_only_in_es] )
204     end
205     batch_start_id = batch_end_id
206   end
207 end

148     end

150     if self.respond_to?(:load_for_index)
151       scope = scope.load_for_index
152     end
153     batch = scope.to_a
154     prepare_for_index(batch)
155     results = elastic_search(
156       sort: { id: :asc },
157       size: options[:batch_size],
158       filters: [
159         { range: { id: { gte: batch_start_id } } },
160         { range: { id: { lt: batch_end_id } } }
161       ],
162       source: ["id"]
163     ).group_by{ |r| r.id.to_i }
164     bulk_index(batch, skip_prepare_batch: true)
165     ids_only_in_es = results.keys - batch.map(&:id)
166     unless ids_only_in_es.empty?
167       Rails.logger.debug "[DEBUG] Deleting vestigial docs in ES: #{ ids_only_in_es }"
168       elastic_delete_by_ids!( [ids_only_in_es] )

209 def result_to_will_paginate_collection(result, options={})
210   try_and_try_again( PG::ConnectionBad, sleep: 20 ) do
211     begin
212       records = options[:keep_es_source] ?
213         result.records.map_with_hit do |record, hit|
214           record.es_source = hit._source
215           record
216         end : result.records.to_a
217       elastic_ids = result.results.results.map{ |r| r.id.to_i }
218       elastic_ids_to_delete = elastic_ids - records.map(&:id)
219       elastic_delete_by_ids!( elastic_ids_to_delete )
220       WillPaginate::Collection.create(result.current_page, result.per_page,
221         result.total_entries - elastic_ids_to_delete.count) do |pager|
222         pager.replace(records)
223       end
224     rescue Elasticsearch::Transport::Transport::Errors::BadRequest => e
225       Logstasher.write_exception(e)
226       Rails.logger.error "[Error] Elasticsearch query failed: #{ e }"
227       Rails.logger.error "Backtrace:\n#{ e.backtrace[0..30].join("\n") }\n..."
228       WillPaginate::Collection.new(1, 30, 0)
229     end
230   end
231 end
