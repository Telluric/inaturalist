144 def picasa_friends(user)
145   return [] unless (pa = current_user.has_provider_auth('google'))
146   picasa = GData::Client::Photos.new
147   picasa.auth_token = pa.token
148   contacts_data = picasa.get("http://picasaweb.google.com/data/feed/api/user/default/contacts").to_xml

150   contacts_data.elements.each('entry'){|e|
151     friends << {
152       'id' => e.elements['gphoto:user'].text, # this is a feed url that id's the photo
153       'name' => e.elements['gphoto:nickname'].text,
154       'pic_url' => e.elements['gphoto:thumbnail'].text
155     }
156   }
