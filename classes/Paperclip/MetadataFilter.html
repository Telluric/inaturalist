   # File lib/paperclip_processors/metadata_filter.rb
15 def make
16   src = @file
17   filename = [@basename, @style, "filtered"].join
18   dst_path = File.join( Dir.tmpdir, filename )
19   begin
20     # Without interpolation, this command looks like this:
21     # exiftool -all= -tagsFromFile @ -ICC_Profile input.jpg -o output.jpg
22     # That should remove all profiles except the ICC color profile
23     Paperclip.run(
24       "exiftool",
25       "-all= -tagsFromFile @ -ICC_Profile :source -o :dest",
26       {
27         source: File.expand_path( src.path ),
28         dest: File.expand_path( dst_path )
29       }
30     )
31   rescue Terrapin::ExitStatusError => e
32     raise Paperclip::Error, "There was an error filtering metadata for #{@basename}: #{e}" if @whiny
33   rescue Terrapin::CommandNotFoundError => e
34     raise Paperclip::Errors::CommandNotFoundError.new( "Could not run the `exiftool` command. Please install exiftool." )
35   end
36   # If there were no errors, we should have a new photo at the dst_path
37   File.open( dst_path, "rb" )
38 end
