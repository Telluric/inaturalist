 80 def self.new_from_api_response(api_response, options = {})
 81   fp = api_response
 82   return nil if fp.nil?
 83   # facebook api provides these sizes in an keyless array, in this order
 84   [
 85     :remote_large_url,
 86     :remote_medium_url,
 87     :remote_small_url,
 88     :remote_thumb_url
 89   ].each_with_index{|img_size,i|
 90     next unless fp['images'] && fp['images'][i]
 91     options.update(img_size => fp['images'][i]['source'])
 92   }
 93   options.update(
 94     :native_photo_id => fp["id"],
 95     :remote_square_url => nil, # facebook doesn't provide a square image
 96     :remote_original_url => fp['source'],
 97     :native_page_url => fp["link"],
 98     :native_username => fp["from"]["name"],
 99     :native_realname => fp["from"]["name"],
100     :license => nil
101   )
102   facebook_photo = FacebookPhoto.new(options)
103   facebook_photo.api_response = fp
104   facebook_photo
105 end
