 72 def run
 73   start_time = Time.now
 74   new_count = 0
 75   old_count = 0
 76   delete_count = 0
 77   count = 0
 78   request
 79   logger.info "[#{Time.now}] Waiting for archive to generate..."
 80   while generating
 81     print "."
 82     sleep 3
 83   end
 84   logger.info
 85   logger.info "[#{Time.now}] Downloading archive..."
 86   download
 87   obs_ids_to_index = []
 88   # "\x00" is an unprintable character that I hope we can assume will never appear in the data. If it does, CSV will choke
 89   CSV.foreach(File.join(@tmp_path, "occurrence.txt"), col_sep: "\t", headers: true, quote_char: "\x00") do |row|
 90     # puts "row['gbifID']: #{row['gbifID']}\t\trow['catalogNumber']: #{row['catalogNumber']}"
 91     observation_id = row['catalogNumber']
 92     gbif_id = row['gbifID']
 93     logger.info [
 94       gbif_id.to_s.ljust(20), 
 95       "#{count} of #{@status['totalRecords']} (#{(count / @status['totalRecords'].to_f * 100).round(2)}%)".ljust(30),
 96       "#{((Time.now - start_time) / 60.0).round(2)} mins"
 97       ].join(' ')
 98     observation = Observation.find_by_id(observation_id)
 99     if observation.blank?
100       logger.info "\tobservation #{observation_id} doesn't exist, skipping..." if @opts[:debug]
101       next
102     end
103     href = "http://www.gbif.org/occurrence/#{gbif_id}"
104     existing = ObservationLink.where(observation_id: observation_id, href: href).first
105     if existing
106       existing.touch unless @opts[:debug]
107       old_count += 1
108       logger.info "\tobservation link already exists for observation #{observation_id}, skipping" if @opts[:debug]
109     else
110       ol = ObservationLink.new( observation: observation, href: href, href_name: "GBIF", rel: "alternate" )
111       ol.save unless @opts[:debug]
112       new_count += 1
113       obs_ids_to_index << observation.id
114       # puts "\tCreated #{ol}"
115     end
116     count += 1
117   end
118 
119   links_to_delete_scope = ObservationLink.where("href_name = 'GBIF' AND updated_at < ?", start_time)
120   delete_count = links_to_delete_scope.count
121   logger.info
122   logger.info "[#{Time.now}] Deleting #{delete_count} observation links..."
123   obs_ids_to_index += links_to_delete_scope.pluck(:observation_id)
124   obs_ids_to_index = obs_ids_to_index.compact.uniq
125   links_to_delete_scope.delete_all unless @opts[:debug]
126 
127   logger.info
128   logger.info "[#{Time.now}] Re-indexing #{obs_ids_to_index.size} observations..."
129   obs_ids_to_index.in_groups_of( 500 ) do |group|
130     Observation.elastic_index!( ids: group.compact, wait_for_index_refresh: true ) unless @opts[:debug]
131     num_indexed += group_size
132     puts "[#{Time.now}] #{num_indexed} re-indexed (#{( num_indexed / obs_ids_to_index.size.to_f * 100 ).round( 2 )})"
133   end
134   logger.info "[#{Time.now}] Finished linking for #{@data_partner}"
135 end
