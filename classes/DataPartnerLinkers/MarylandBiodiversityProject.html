 3 def run
 4   url = "https://www.marylandbiodiversity.com/services/iNatMBPIDMap.php"
 5   page = 1
 6   start_time = Time.now
 7   obs_ids_to_index = []
 8   per_page = 500
 9   new_count = 0
10   old_count = 0
11   while true
12     logger.info "Requesting #{url}, page: #{page}"
13     resp = RestClient.get( url, params: { page: page, per_page: per_page } )
14     json = JSON.parse( resp )
15     break if json.size == 0
16     obs_ids = json.map{|r| r["inat_id"] }
17     existing_observation_links = ObservationLink.where(
18       observation_id: obs_ids,
19       href_name: @data_partner.name
20     ).to_a
21     observations = Observation.page_of_results( id: obs_ids, per_page: per_page )
22     logger.debug "#{obs_ids.size} obs IDs in MBP batch #{page} starting with #{obs_ids.first}"
23     logger.debug "#{existing_observation_links.size} existing links"
24     logger.debug "#{observations.size} obs extant"
25     json.each do |row|
26       partner_record_id = row["mbp_id"]
27       observation_id = row["inat_id"]
28       href = "https://www.marylandbiodiversity.com/record/#{partner_record_id}"
29       # existing = ObservationLink.where( observation_id: observation_id, href: href ).first
30       observation = observations.detect{|o| o.id == observation_id.to_i }
31       if existing = existing_observation_links.detect{|ol| ol.href == href }
32         existing.touch unless @opts[:debug]
33         old_count += 1
34         # logger.debug "\tobservation link already exists, skipping"
35         if observation && observation.last_indexed_at < existing.created_at
36           obs_ids_to_index << observation.id
37         end
38       elsif observation
39         ol = ObservationLink.new(
40           observation: observation,
41           href: href,
42           href_name: @data_partner.name,
43           rel: "alternate"
44         )
45         ol.save unless @opts[:debug]
46         new_count += 1
47         obs_ids_to_index << observation.id
48         logger.debug "\tCreated #{ol}"
49       end
50     end
51     page += 1
52     sleep 5
53   end
54   links_to_delete_scope = ObservationLink.where( "href_name = ? AND updated_at < ?", @data_partner.name, start_time )
55   delete_count = links_to_delete_scope.count
56   obs_ids_to_index += links_to_delete_scope.pluck(:observation_id)
57   logger.info "Deleting #{delete_count} links..."
58   links_to_delete_scope.delete_all unless @opts[:debug]
59   logger.info "Re-indexing #{obs_ids_to_index.size} observations..."
60   obs_ids_to_index = obs_ids_to_index.compact.uniq
61   obs_ids_to_index_groups = obs_ids_to_index.in_groups_of( 500 )
62   obs_ids_to_index_groups.each_with_index do |group,i|
63     logger.info "Indexing obs group #{i} / #{obs_ids_to_index_groups.size}"
64     Observation.elastic_index!( ids: group.compact )
65   end
66   logger.info ""
67   logger.info "MarylandBiodiversityProject created #{new_count} new links, #{old_count} existing, #{delete_count} deleted, #{obs_ids_to_index.size} obs reindexed"
68   logger.info ""
69 end
