Constants

MAX_DOWNLOAD_LINES = 50000
 

This is the maximum number of lines Calflora will return in a download. If we get a response this size or bigger, we need to partition the data requesting to get smaller responses so we don't miss anything

53 def partition_data(after = Date.parse("1900-01-01"), before=Date.today)
54   rows = get_calflora_rows( after, before )
55   if rows.size >= MAX_DOWNLOAD_LINES
56     split = before - ( ( before - after ) / 2 )
57     partition_data( after, split )
58     partition_data( split, before )
59   else
60     process_rows( rows )
61   end
62 end

 64 def process_rows( rows )
 65   logger.info "Processing #{rows.size} rows"
 66   per_page = 300
 67   rows.in_groups_of( per_page ) do |grp|
 68     pairs = grp.compact.map do |row|
 69       calflora_id, source = row
 70       next if calflora_id.blank?
 71       observation_id = calflora_id[/in:(.+)/, 1]
 72       if observation_id.blank? && !source.blank?
 73         observation_id = source[/inaturalist\.org\/observations\/([^\/\?]+)/, 1]
 74       end
 75       [calflora_id, observation_id]
 76     end
 77     obs_ids = pairs.compact.map(&:last).compact
 78     observations = Observation.page_of_results( id: obs_ids, per_page: per_page ).index_by(&:id)
 79     observation_links = ObservationLink.where( observation_id: obs_ids ).group_by(&:observation_id)
 80     pairs.each do |calflora_id, observation_id|
 81       next unless calflora_id
 82       logger.debug calflora_id
 83       observation = observations[observation_id.to_i]
 84       if observation.blank?
 85         logger.debug "\tobservation #{observation_id} doesn't exist, skipping..."
 86         next
 87       end
 88       href = "http://www.calflora.org/cgi-bin/noccdetail.cgi?seq_num=#{calflora_id}"
 89       existing = ( observation_links[observation_id.to_i] || [] ).detect{|ol| ol.href == href}
 90       if existing
 91         existing.touch unless @opts[:debug]
 92         @old_count += 1
 93         logger.debug "\tobservation link already exists, skipping"
 94       else
 95         ol = ObservationLink.new(
 96           observation: observation,
 97           href: href,
 98           href_name: @data_partner.name,
 99           rel: "alternate"
100         )
101         unless ol.valid?
102           logger.error "#{ol} not valid: #{ol.errors.full_messages.to_sentence}"
103         end
104         ol.save unless @opts[:debug]
105         @new_count += 1
106         @obs_ids_to_index << observation.id
107         logger.debug "\tCreated #{ol}"
108       end
109     end
110   end
111 end
