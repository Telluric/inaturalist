168 def queries
169   replica_pool = ActiveRecord::Base.connection.instance_variable_get( "@replica_pool" )
170   if replica_pool
171     # if configured to use replica DBs with Makara, fetch queries from all primaries and replicas
172     primary_pool = ActiveRecord::Base.connection.instance_variable_get( "@primary_pool" )
173     @queries = []
174     ( primary_pool.connections + replica_pool.connections ).flatten.each do |connection|
175       instance_queries = connection.active_queries.map do |q|
176         { db_host: connection.config[:host] }.merge( q )
177       end
178       @queries += instance_queries
179     end
180   else
181     @queries = ActiveRecord::Base.connection.active_queries.map do |q|
182       { db_host: ActiveRecord::Base.connection_db_config.host }.merge( q )
183     end
184   end
185   @queries.delete_if{ |q| q["query"] =~ /pg_stat_activity/ }
186   render layout: "admin"
187 end
