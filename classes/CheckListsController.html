17 def show
18   @place = @list.place
19   @other_check_lists = @place.check_lists.where("id != ?", @list.id).limit(1000)
20 
21   # If this is a place's default check list, load ALL the listed taxa
22   # belonging to this place.  Kind of weird, I know.  The alternative would
23   # be to keep the default list updated with duplicates from all the other
24   # check lists belonging to a place, like we do with parent lists.  It
25   # would be a pain to manage, but it might be faster.
26   if @list.is_default?
27     @unpaginated_listed_taxa = ListedTaxon.find_listed_taxa_from_default_list(@list.place_id)
28 
29     # Searches must use place_id instead of list_id for default checklists
30     # so we can search items in other checklists for this place
31     if (@q = params[:q]) && !@q.blank?
32       @search_taxon_ids = Taxon.elastic_search(
33         filters: [ {
34           nested: {
35             path: "names",
36             query: {
37               match: { "names.name": { query: @q, operator: "and" } }
38             }
39           }
40         } ]
41       ).per_page(1000).map(&:id)
42       @unpaginated_listed_taxa = @unpaginated_listed_taxa.filter_by_taxa(@search_taxon_ids)
43     end
44   end
45 
46   if params[:find_missing_listings]
47     @missing_filter_taxon = params[:missing_filter_taxon].present? ? Taxon.find(params[:missing_filter_taxon]) : nil
48     @hide_descendants = params[:hide_descendants]
49     @hide_ancestors = params[:hide_ancestors]
50     @missing_listings_list = params[:missing_listing_list_id].present? ? List.find_by_id(params[:missing_listing_list_id]) : nil
51     list_ids_from_projects = ProjectList.joins(:project).where("projects.place_id = ?", @list.place_id).pluck(:id)
52     @lists_for_missing_listings = List.where("(place_id = ? AND id != ?) OR id IN (?)", @list.place_id, @list.id, list_ids_from_projects).order(:title)
53     missing_listings_list_ids = @lists_for_missing_listings.map(&:id)
54     listed_taxa_on_this_list = @list.find_listed_taxa_and_ancestry_as_hashes
55     listed_taxa_on_other_lists = @list.find_listed_taxa_and_ancestry_on_other_lists_as_hashes(missing_listings_list_ids)
56     scoped_list = apply_missing_listings_scopes(listed_taxa_on_this_list, listed_taxa_on_other_lists, @missing_filter_taxon, @hide_ancestors, @hide_descendants, @missing_listings_list)
57     ids_for_listed_taxa_on_other_lists = scoped_list.map{|lt| lt['id'] }
58     @missing_listings = ListedTaxon.where('listed_taxa.id IN (?)', ids_for_listed_taxa_on_other_lists).paginate({:page => params[:missing_listings_page] || 1, :per_page => 20})
59   end
