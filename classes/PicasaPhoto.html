 96 def self.get_api_response(native_photo_id, options = {})
 97   # Picasa API calls only work with a user's token, so first we try to get
 98   # a PicasaIdentity from a passed in user, then we try to parse one out of
 99   # the native ID (which should be a URL)
100   picasa_identity = if options[:user]
101     options[:user].picasa_identity
102   elsif native_photo_id.is_a?(String) && matches = native_photo_id.match(/user\/(.+?)\//)
103     User.find_by_id(matches[1]).try(:picasa_identity)
104   else
105     nil
106   end
107   
108   if picasa_identity
109     PicasaPhoto.picasa_request_with_refresh( picasa_identity ) do
110       goog = GooglePhotosApi.new( picasa_identity.token )
111       goog.media_item( native_photo_id )
112     end
113   else
114     nil
115   end
116 end

61 def self.repair(find_options = {})
62   puts "[INFO #{Time.now}] starting PicasaPhoto.repair, options: #{find_options.inspect}"
63   find_options[:include] ||= [:user, :taxon_photos, :observation_photos]
64   find_options[:batch_size] ||= 100
65   find_options[:sleep] ||= 10
66   updated = 0
67   destroyed = 0
68   invalids = 0
69   skipped = 0
70   start_time = Time.now
71   PicasaPhoto.where(find_options).find_each do |p|
72     r = Net::HTTP.get_response(URI.parse(p.medium_url))
73     unless [Net::HTTPBadRequest, Net::HTTPForbidden, Net::HTTPRedirection].include?(r.code_type)
74       skipped += 1
75       next
76     end
77     repaired, errors = p.repair
78     if errors.blank?
79       updated += 1
80     else
81       puts "[ERROR] #{errors.values.to_sentence}"
82       if repaired.frozen?
83         destroyed += 1 
84         puts "[ERROR] destroyed #{repaired}"
85       end
86       if errors[:picasa_account_not_linked]
87         invalids += 1
88         puts "[ERROR] Picasa account not linked for #{repaired}"
89       end
90     end
91   end
92   puts "[INFO #{Time.now}] finished PicasaPhoto.repair, #{updated} updated, #{destroyed} destroyed, #{invalids} invalid, #{skipped} skipped, #{Time.now - start_time}s"

11 def to_observation  
12   self.api_response ||= PicasaPhoto.get_api_response(self.native_photo_id, :user => self.user)
13   
14   # Setup the observation
15   observation = Observation.new
16   observation.user = self.user if self.user
17   observation.observation_photos.build(:photo => self, :observation => observation)
18   observation.description = api_response["description"]
19   if timestamp = api_response["creationTime"]
20     # Google seems to present these as UTC these days (I think)
21     observation.observed_on_string = ActiveSupport::TimeZone.new('UTC').at(timestamp / 1000).to_s(:long)
22   end
23   observation.munge_observed_on_with_chronic
24   observation.time_zone = observation.user.time_zone if observation.user
25   
26   # As of Feb 2019, the Picasa API went away and was replaced with the Google
27   # Photos API, which doesn't return any geodata
28   
29   # Try to get a taxon
30   observation.taxon = to_taxon
31   if t = observation.taxon
32     observation.species_guess = t.common_name.try(:name) || t.name
33   end
34
