35 def commit
36   super
37   input_taxa.each do |input_taxon|
38     #duplicate photos
39     input_taxon.taxon_photos.sort_by(&:id).each do |taxon_photo|
40       begin
41         output_taxon.photos << taxon_photo.photo
42       rescue ActiveRecord::RecordInvalid => e
43         Rails.logger.error "[ERROR #{Time.now}] Failed to add #{taxon_photo} to #{output_taxon}: #{e}"
44       end
45     end
46     
47     #duplicate taxon_names
48     input_taxon.taxon_names.each do |taxon_name|
49       taxon_name.reload
50       unless taxon_name.valid?
51         Rails.logger.info "[INFO] Destroying #{taxon_name} while committing taxon change " +
52           "#{self.id}: #{taxon_name.errors.full_messages.to_sentence}"
53         taxon_name.destroy
54         next
55       end
56       new_taxon_name = taxon_name.dup
57       new_taxon_name.taxon_id = output_taxon.id
58       new_taxon_name.is_valid = false if taxon_name.is_scientific_names? && taxon_name.is_valid?
59       unless new_taxon_name.save
60         Rails.logger.error "[ERROR #{Time.now}] TaxonChange #{id} failed to duplicate #{taxon_name}: " + 
61           new_taxon_name.errors.full_messages.to_sentence
62       end
63     end
64     
65     #duplicate colors
66     output_taxon.colors << input_taxon.colors if output_taxon.colors.blank?
67 
68     # Move input child taxa to the output taxon
69     delay( priority: USER_PRIORITY ).move_input_children_to_output( input_taxon.id )
70   end
71 end
