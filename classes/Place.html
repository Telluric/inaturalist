import_from_shapefile(shapefile_path, options = {}, &block) Link

Import places from a shapefile. Note that this is optimized for use with a set of adapter methods in PlaceSources. Note that this always assumes shapefiles have a geographic projection using a NAD83 / WGS84 datum and lat/lon coordinates. Options:

<tt>source</tt>: specify a type of handler for certain shapefiles.  Current options are 'census', 'esriworld', and 'cpad'
<tt>test</tt>: (boolean) setting this to +true+ will do everything other than saving places and geometries.
<tt>ancestor_place</tt>: (Place) scope searches for exissting records to descendents of this place. Matching will be based on name and place_type
<tt>name_column</tt>: column in shapefile attributes that holds the name of the place
<tt>source_identifier_column</tt>: column in shapefile attributes that holds a unique identifier

Examples:

Census:
  Place.import_from_shapefile('/Users/kueda/Desktop/tl_2008_06_county/tl_2008_06_county.shp', :place_type => 'county', :source => 'census')

California Protected Areas Database:
  Place.import_from_shapefile('/Users/kueda/Desktop/CPAD_March09/Units_Fee_09_longlat.shp', :source => 'cpad')

546   start_time = Time.now
547   num_created = num_updated = 0
548   src = options[:source]
549   options.delete(:source) unless src.is_a?(Source)
550   RGeo::Shapefile::Reader.open( shapefile_path ) do |file|
551     file.each do |shp|
552       puts "[INFO] Working on shp..."
553       new_place = case src
554       when 'census'
555         PlaceSources.new_place_from_census_shape(shp, options)
556       when 'esriworld'
557         PlaceSources.new_place_from_esri_world_shape(shp, options)
558       when 'cpad'
559         PlaceSources.new_place_from_cpad_units_fee(shp, options)
560       else
561         Place.new_from_shape(shp, options)
562       end
563       
564       unless new_place
565         puts "[INFO] \t\tShape couldn't be converted to a place.  Skipping..."
566         next
567       end
568       
569       new_place.source_filename = options[:source_filename] || File.basename(shapefile_path)
570       new_place.source ||= src if src.is_a?(Source)
571         
572       puts "[INFO] \t\tMade new place: #{new_place}"
573       
574       # Try to find an existing place
575       existing = nil
576       if !new_place.source_filename.blank? && !new_place.source_identifier.blank?
577         existing ||= Place.where(source_filename: new_place.source_filename,
578           source_identifier: new_place.source_identifier).first
579       end
580       if !new_place.source_filename.blank? && !new_place.source_name.blank?
581         existing ||= Place.where(source_filename: new_place.source_filename,
582           source_name: new_place.source_name).first
583       end
584       if !new_place.source_filename.blank? && !new_place.name.blank?
585         existing ||= begin
586           Place.where(source_filename: new_place.source_filename).where("lower(name) = ?", new_place.name.downcase).first
587         rescue
588           new_place.name = new_place.name.force_encoding('ISO-8859-1').encode('UTF-8')
589           Place.where(source_filename: new_place.source_filename).where("lower(name) = ?", new_place.name.downcase).first
590         end
591       end
592       if options[:ancestor_place]
593         existing ||= options[:ancestor_place].descendants.
594           where("lower(name) = ? AND place_type = ?", new_place.name.downcase, new_place.place_type).first
595       end
596       
597       if existing
598         puts "[INFO] \t\tFound existing place: #{existing}"
599         place = existing
600         [:swlat, :swlng, :nelat, :nelng, :source_filename, :source_name, 
601             :source_identifier].each do |attr_name|
602           place.send("#{attr_name}=", new_place.send(attr_name)) if new_place.send(attr_name)
603         end
604         num_updated += 1
605       else
606         place = new_place
607         num_created += 1
608       end
609 
610       if options[:ancestor_place]
611         place.parent ||= options[:ancestor_place]
612       end
613 
614       place.place_type = options[:place_type] unless options[:place_type].blank?
615       place.place_type_name = options[:place_type_name] unless options[:place_type_name].blank?
616       
617       place = if block_given?
618         yield place, shp
619       else
620         place
621       end
622       begin
623         if place && place.valid?
624           place.save! unless options[:test]
625           puts "[INFO] \t\tSaved place: #{place}, parent: #{place.parent.try(:name)}"
626         else
627           num_created -= 1
628           puts "[ERROR] \tPlace invalid: #{place.errors.full_messages.join(', ')}" if place
629           next
630         end
631       rescue ArgumentError => e
632         raise e unless e.message =~ /Cannot transliterate/
633         # Pretend it's actually UTF
634         place.name.force_encoding( "UTF-8" )
635         retry
636       rescue => e
637         puts "[ERROR] \tError: #{e}"
638         next
639       end
640       
641       next if options[:test]
642       
643       if existing && PlaceGeometry.exists?(
644           ["place_id = ? AND updated_at >= ?", existing, start_time.utc])
645         puts "[INFO] \t\tAppending to existing geom..."
646         place.append_geom(shp.geometry, :source => options[:source])
647       else
648         puts "[INFO] \t\tAdding geom..."
649         place.save_geom(shp.geometry, 
650           :source => options[:source],
651           :source_filename => place.source_filename,
652           :source_name => place.source_name, 
653           :source_identifier => place.source_identifier)
654       end
655       place.place_geometry_without_geom.process_geometry
656     end
657   end
658   
659   puts "\n[INFO] Finished importing places.  #{num_created} created, " + 
660     "#{num_updated} updated (#{Time.now - start_time}s)"

829   return false unless bounding_box

    # File app/models/place.rb
691   def merge(mergee, options = {})
692     # Keep chosen attributes from the mergee
693     if keepers = options[:keep]
694       keepers =  [options[:keep]] unless keepers.is_a?(Array)
695       keepers.each do |attr_name|
696         self.send("#{attr_name}=", mergee.send(attr_name))
697       end
698     end
699     
700     # Hack!  We want to make sure the updates don't invalidate the place
701     # BEFORE we moving stuff around.  This hackery gets us around the unique
702     # name within a parent validation
703     temp_name = self.name
704     self.name += '_'
705     unless self.valid?
706       self.name = temp_name
707       return self
708     end
709     self.name = temp_name
710     
711     # Move the mergee's listed_taxa to the target's default check list
712     mergee.check_lists.each do |cl|
713       cl.update( place: self )
714       ListedTaxon.where( list_id: cl.id ).update_all( place_id: id )
715     end
716     if check_list
717       ListedTaxon.where( list_id: mergee.check_list_id ).update_all( list_id: check_list_id, place_id: id )
718     elsif mergee.check_list && mergee.check_list.listed_taxa.count > 0
719       mergee.check_list.update( place_id: id, title: "MERGED #{mergee.check_list.title}")
720       ListedTaxon.where( list_id: mergee.check_list_id ).update_all( place_id: id )
721     end
722     
723     # Keep reject geometry if keeper doesn't have one
724     if place_geometry_without_geom.nil? && !mergee.place_geometry_without_geom.nil?
725       save_geom(mergee.place_geometry.geom)
726     end
727 
728     mergee.children.each do |child|
729       child.parent = self
730       unless child.save
731         # If there's a problem saving the child, orphan it. Otherwise it will
732         # get deleted when the parent is deleted
733         child.update(:parent => nil)
734       end
735     end
736 
737     # Add any observations_places from mergee into this place
738     # which didn't aleady exist
739     Place.connection.execute <<-SQL
740       INSERT INTO observations_places (observation_id, place_id)
741       SELECT op.observation_id, #{id} FROM observations_places op
742       WHERE place_id = #{mergee.id}
743       AND NOT EXISTS (
744         SELECT observation_id FROM observations_places
745         WHERE place_id = #{id} AND observation_id = op.observation_id
746       )
747     SQL
748     # Because of the above, if the mergee had any observations_places
749     # those merge queries will fail here and the observations_places will
750     # remain associated with mergee. This is good because on mergee destroy,
751     # that will trigger Place.update_observations_places(mergee) which
752     # will index all the observations that used to be in mergee
753     merge_has_many_associations(mergee)
754     
755     # ensure any loaded associates that had their foreign keys updated in the db aren't hanging around
756     mergee.reload
757     mergee.destroy
758     self.save
759     CheckList.where(place_id: id).each do |cl|
760       cl.delay(priority: USER_INTEGRITY_PRIORITY,
761         unique_hash: { "CheckList::refresh": cl.id }
762       ).refresh
763     end
764     if self.check_list
765       self.check_list.delay(priority: USER_INTEGRITY_PRIORITY,
766         unique_hash: { "CheckList::refresh": self.check_list.id }
767       ).refresh
768     end
769     self
770   end

438 def validate_with_geom( geom, other_attrs = {} )
439   if geom.is_a?( GeoRuby::SimpleFeatures::Geometry )
440     georuby_geom = geom
441     geom = RGeo::WKRep::WKBParser.new.parse( georuby_geom.as_wkb ) rescue nil
442   end
443   if geom.blank?
444     # This probably means GeoRuby parsed some polygons but RGeo didn't think
445     # they looked like a multipolygon, possibly because of overlapping

451   if other_attrs[:user] && !other_attrs[:user].is_admin?

453        ( CONFIG.content_freeze_enabled && geom.area > MAX_PLACE_AREA_FOR_NON_STAFF_DURING_FREEZE ) ||

458       observation_count >= MAX_PLACE_OBSERVATION_COUNT )
459       add_custom_error(:place_geometry, :contains_too_many_observations)

461     end
462   end
