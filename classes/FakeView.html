61 def place_geometry_kml_url( options = {} )
62   place = options[:place] || @place
63   return "" if place.blank?
64 
65   place_geometry = options[:place_geometry]
66   place_geometry ||= place.place_geometry_without_geom if place.association( :place_geometry_without_geom ).loaded?
67   place_geometry ||= place.place_geometry if place.association( :place_geometry ).loaded?
68   place_geometry ||= PlaceGeometry.without_geom.where( place_id: place ).first
69   if place_geometry.blank?
70     "".html_safe
71   else
72     "#{place_geometry_url( place, format: "kml" )}?#{place_geometry.updated_at.to_i}".html_safe
73   end
74 end
