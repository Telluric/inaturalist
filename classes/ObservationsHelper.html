 2 def observation_image_url(observation, params = {})
 3   return nil if observation.observation_photos.blank?
 4   size = params[:size].blank? ? "square" : params[:size]
 5   photo = observation.observation_photos.sort_by do |op|
 6     op.position || observation.observation_photos.size + op.id.to_i
 7   end.first.photo
 8   url = photo.best_url( size )
 9   return nil if !url
10   # this assumes you're not using SSL *and* locally hosted attachments for observations
11   if params[:ssl] || ( defined?( request ) && request && request.protocol =~ /https/ )
12     url = url.sub("http://", "https://s3.amazonaws.com/")
13   end
14   url
15 end

21 def observations_order_by_options(order_by = nil)
22   order_by ||= @order_by
23   fields = ObservationsController::ORDER_BY_FIELDS - %w(species_guess)
24   pairs = fields.map do |f|
25     next if f == "project" && @project.blank?
26     value = %w(created_at observations.id id).include?(f) ? "observations.id" : f
27     default = ObservationsController::DISPLAY_ORDER_BY_FIELDS[f].to_s
28     key = default.parameterize.underscore
29     key_capitalized = "#{key}_"
30     [t( key_capitalized, default: t( key, default: default ) ), value]
31   end.compact
32   order_by = "observations.id" if order_by.blank?
33   options_for_select(pairs, order_by)
34 end
