Set instance variables on an instance that acts_as_spammable will use when checking with akismet. It seems like user_ip is the only one that's really required, which we set for all users anyway, so this probably isn't necessary, but it would probably help.

68 def check_spam( options = {} )
69   before_action :set_akismet_params_on_record, only: options[:only], except: options[:except]
70 
71   define_method( :set_akismet_params_on_record ) do
72     return unless record = instance_variable_get( "@" + options[:instance].to_s )
73     # if we checking for spam when creating / updating a user and the
74     # current user is not that user (e.g. curator or admin action), we
75     # don't want to send akismet the current user's info
76     if record.is_a?( User ) && current_user && current_user != record
77       return
78     end
79     # No point in setting this info if we know this user isn't a spammer
80     return if current_user && current_user.known_non_spammer?
81     record.acts_as_spammable_user_ip = Logstasher.ip_from_request_env( request.env )
82     record.acts_as_spammable_user_agent = request.user_agent
83     record.acts_as_spammable_referrer = request.referrer
84   end
