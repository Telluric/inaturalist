photo_fields() Link

Return an HTML fragment containing checkbox inputs for Flickr photos. Params:

q:        query string
context:  Set to 'user' to search only the logged in user's photos. 
          Otherwise search all CC'd photos.
index:    index to use if these inputs are part of a form subcomponent
          (makes names like flickr_photos[:index][])

 7 def photos
 8   f = get_flickraw
 9   params[:limit] ||= 10
10   params[:page] ||= 1
11   unless params[:q]
12     @photos = f.photos.search(
13       :user_id => current_user.flickr_identity.flickr_user_id, 
14       :extras => 'url_s',
15       'per_page' => params[:limit], 
16       'page' => params[:page])
17   else
18     # Try to look up a photo id
19     if params[:q].to_i != 0 && params[:q].to_i > 10000
20       @photos = [f.photos.getInfo(:photo_id => params[:q])].compact
21     else
22       @photos = @flickr.photos.search(
23           'user_id' => current_user.flickr_identity.flickr_user_id, 
24           'text' => "#{params[:q]}",
25           :extras => 'url_s',
26           'per_page' => params[:limit],
27           'page' => params[:page]).map do |fp|
28         FlickrPhoto.new_from_api_response(fp)
29       end
30     end
31   end
32   respond_to do |format|
33     format.html
34     format.json { render :json => @photos.to_json }
35   end
