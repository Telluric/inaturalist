 57 def commit
 58   super
 59   # duplicate photos
 60   input_taxon.taxon_photos.sort_by(&:id).each do |taxon_photo|
 61     begin
 62       output_taxon.photos << taxon_photo.photo
 63     rescue ActiveRecord::RecordInvalid => e
 64       Rails.logger.error "[ERROR #{Time.now}] Failed to add #{taxon_photo} to #{output_taxon}: #{e}"
 65     end
 66   end
 67 
 68   # duplicate conservation_statuses
 69   input_taxon.conservation_statuses.each do |cs|
 70     new_cs = cs.dup
 71     new_cs.taxon_id = output_taxon.id
 72     unless new_cs.save
 73       Rails.logger.error "[ERROR #{Time.now}] TaxonChange #{id} failed to duplicate #{cs}: " + 
 74         new_cs.errors.full_messages.to_sentence
 75     end
 76   end
 77   
 78   # duplicate taxon_names
 79   input_taxon.taxon_names.each do |taxon_name|
 80     taxon_name.reload
 81     unless taxon_name.valid?
 82       Rails.logger.info "[INFO] Destroying #{taxon_name} while committing taxon change " +
 83         "#{self.id}: #{taxon_name.errors.full_messages.to_sentence}"
 84       taxon_name.destroy
 85       next
 86     end
 87     new_taxon_name = taxon_name.dup
 88     new_taxon_name.creator = nil
 89     new_taxon_name.updater = nil
 90     if new_taxon_name.source_url.blank?
 91       new_taxon_name.source_url = FakeView.edit_taxon_name_url( taxon_name )
 92     end
 93     if new_taxon_name.source_identifier.blank?
 94       new_taxon_name.source_identifier = taxon_name.id
 95     end
 96     new_taxon_name.taxon_id = output_taxon.id
 97     new_taxon_name.is_valid = false if taxon_name.is_scientific_names? && taxon_name.is_valid?
 98     unless new_taxon_name.save
 99       Rails.logger.error "[ERROR #{Time.now}] TaxonChange #{id} failed to duplicate #{taxon_name}: " + 
100         new_taxon_name.errors.full_messages.to_sentence
101     end
102   end
103   
104   # duplicate taxon_range
105   if output_taxon.taxon_range.nil?
106     if taxon_range = input_taxon.taxon_range
107       new_taxon_range = taxon_range.dup
108       new_taxon_range.taxon_id = output_taxon.id
109       unless new_taxon_range.save
110         Rails.logger.error "[ERROR #{Time.now}] TaxonChange #{id} failed to duplicate #{taxon_range}: " +
111           new_taxon_range.errors.full_messages.to_sentence
112       end
113       unless new_taxon_range.create_kml_attachment
114         Rails.logger.error "[ERROR #{Time.now}] TaxonChange #{id} failed to duplicate #{taxon_range} attachment: " +
115           new_taxon_range.errors.full_messages.to_sentence
116       end
117     end
118   end
119   
120   # duplicate atlas
121   unless output_taxon.atlas
122     if atlas = input_taxon.atlas
123       new_atlas = atlas.dup
124       new_atlas.taxon_id = output_taxon.id
125       unless new_atlas.save
126         Rails.logger.error "[ERROR #{Time.now}] Atlas #{id} failed to duplicate #{atlas}: " +
127           new_atlas.errors.full_messages.to_sentence
128       end
129     end
130   end
131   
132   # duplicate colors
133   output_taxon.colors << input_taxon.colors if output_taxon.colors.blank?
134   
135   # Move input child taxa to the output taxon
136   delay( priority: USER_PRIORITY ).move_input_children_to_output( input_taxon.id )
137 end
