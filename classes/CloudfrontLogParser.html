63 def convert_cloudfront_log( unzipped_path, converted_path )
64   headers = parse_headers( unzipped_path )
65   return if headers.blank?
66   cloudfront_json_file = File.open( converted_path, "w" )
67   CSV.foreach( unzipped_path, headers: headers, col_sep: "\t" ) do |row|
68     if row["sc-bytes"] && row["c-ip"] && row["x-edge-response-result-type"] != "Error"
69       size = nil
70       if matches = row["cs-uri-stem"].match( /^\/photos\/[0-9]+\/([a-z]+)\./ )
71         size = matches[1]
72       end
73       payload = {
74         timestamp: "#{row['date']}T#{row['time']}Z",
75         result_type: row["x-edge-response-result-type"],
76         clientip: row["c-ip"],
77         bytes: row["sc-bytes"].to_i,
78         status_code: row["sc-status"],
79         referrer: row["cs(Referer)"],
80         file_path: row["cs-uri-stem"],
81         image_version: size
82       }
83       cloudfront_json_file.write( "#{payload.to_json}\n" )
84     end
85   end
86   cloudfront_json_file.close
87   true

25 def download_cloudfront_log( log_object )
26   return unless log_object && log_object.is_a?( Aws::S3::Types::Object )
27   basedir = "/tmp/#{LOG_BUCKET_NAME}"
28   FileUtils.mkdir_p( "/tmp/#{LOG_BUCKET_NAME}" )
29   zipped_path = File.join( basedir, log_object.key )
30   unzipped_path = File.join( basedir, log_object.key.sub( ".gz", "" ) )
31   converted_path = File.join( basedir, log_object.key.sub( ".gz", ".json" ) )
32   FileUtils.rm( zipped_path ) rescue nil
33   FileUtils.rm( unzipped_path ) rescue nil
34   fetched_object = s3_client.get_object( {
35     bucket: LOG_BUCKET_NAME,
36     key: log_object.key
37   }, target: zipped_path ) rescue nil
38   if fetched_object && fetched_object.body && File.exists?( zipped_path )
39     system( "gunzip #{zipped_path}" )
40     if File.exists?( unzipped_path )
41       convert_cloudfront_log( unzipped_path, converted_path )
42       if File.exists?( converted_path )
43         s3_client.delete_object( bucket: LOG_BUCKET_NAME, key: log_object.key )
44       end
45     end
46   end
47   true
48 end

50 def parse_headers( unzipped_path )
51   headers = []
52   rownum = 0
53   CSV.foreach( unzipped_path ) do |row|
54     if rownum == 1
55       headers = row[0].gsub( /#Fields:/, "" ).strip.split( " " )
56     end
57     break if rownum >= 1
58     rownum += 1
59   end
60   headers
