115 def for_each_observation( search_params )
116   batch_i = 0
117   obs_i = 0
118   site = user.site || Site.default
119   Observation.search_in_batches( search_params ) do |batch|
120     if @debug
121       logger.info
122       logger.info "BATCH #{batch_i}"
123       logger.info
124     end
125     unless FlowTask.where( id: id ).exists?
126       raise ObservationsExportDeleted.new( "Export was deleted during its run" )
127     end
128     Observation.preload_associations(batch, preloads)
129     batch.each do |observation|
130       logger.info "Obs #{obs_i} (#{observation.id})" if @debug
131       observation.localize_locale = user.locale || site.locale
132       observation.localize_place = user.place || site.place
133       yield observation
134       obs_i += 1
135       logger.info "Finished Obs #{obs_i} (#{observation.id})" if @debug
136     end
137     batch_i += 1
138   end
139 end

42 def run(run_options = {})
43   @logger = run_options[:logger] if run_options[:logger]
44   @debug = run_options[:debug]
45   unless persisted?
46     raise ObservationsExportNotSaved.new( "Export must be saved before being run" )
47   end
48   update(finished_at: nil, error: nil, exception: nil)
49   outputs.each(&:destroy)
50   outputs.reload
51   query = inputs.first.extra[:query]
52   format = options[:format]
53   # if using makara, release stuck primary connections as
54   # the following queries can and should be run against replicas
55   if ActiveRecord::Base.connection.respond_to?(:enable_context_refresh)
56     Makara::Context.release_all
57   end
58   archive_path = case format
59   when 'json'
60     json_archive
61   else
62     csv_archive
63   end
64   logger.info "ObservationsExportFlowTask #{id}: Created archive at #{archive_path}" if @debug
65   open(archive_path) do |f|
66     self.outputs.create!(:file => f)
67   end
68   logger.info "ObservationsExportFlowTask #{id}: Created outputs" if @debug
69   # The user may have requested an email confirmation after this run started
70   reload
71   if options[:email]
72     Emailer.observations_export_notification(self).deliver_now
73     logger.info "ObservationsExportFlowTask #{id}: Emailed user #{user_id}" if @debug
74   end
75   true
76 rescue Exception => e
77   exception_string = [ e.class, e.message ].join(" :: ")
78   logger.error "ObservationsExportFlowTask #{id}: Error: #{exception_string}" if @debug
79   update(
80     finished_at: Time.now,
81     error: "Error",
82     exception: [ exception_string, e.backtrace ].join("\n")
83   )
84   # The user may have requested an email confirmation after this run started
85   reload
86   if options[:email]
87     Emailer.observations_export_failed_notification(self).deliver_now
88     logger.error "ObservationsExportFlowTask #{id}: Emailed user #{user_id} about error" if @debug
89   end
90   false
