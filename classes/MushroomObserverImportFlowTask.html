46 def get( url, params = {} )
47   try = params.delete( :try ) || 1
48   resp = begin
49     log "GET #{url}, #{params}"
50     RestClient.get( url, params: params, "User-Agent" => "iNaturalist/MushroomObserverImportFlowTask" )
51   rescue Net::ReadTimeout, RestClient::Exceptions::ReadTimeout
52     raise TimeoutError,
53       "Request timed out. You might want to let Mushroom Observer know that #{url} may not be working"
54   rescue RestClient::TooManyRequests
55     OpenStruct.new( code: 429 )
56   end
57   if resp.code == 429
58     log "429 error from Mushroom Observer for #{url}, try #{try}"
59     if tries > 3
60       raise TooManyRequestsError,
61         "iNat made too many requests to Mushroom Observer so MO stopped returning our calls"
62     end
63     sleep try * 30 # seconds
64     resp = get( url, params.merge( try: try + 1 ) )
65   end
66   resp
67 end

205 def observation_from_result( result, options = {} )
206   mo_url = mo_url_from_result( result )
207   # Ensure there's actually response to this URL and it's not an error other
208   # than Unauthorized or Forbidden
209   unless ( h = fetch_head( mo_url ) ) && ( h.code.to_i < 400 || [401, 403].include?( h.code.to_i ) )
210     warn( mo_url, "Mushroom Observer is not showing an observation for this URL" )
211     return nil
212   end
213   log "working on result #{mo_url}"
214   if ( is_collection_location = result.at( "is_collection_location" ) ) && is_collection_location[:value] == "false"
215     warn( mo_url, "Obs not from collection location, skipped" )
216     return nil
217   end
218   existing = Observation.joins( :observation_field_values ).
219     where(
220       "observation_field_values.observation_field_id = ? AND value = ?",
221       mo_url_observation_field,
222       mo_url
223     ).first
224   if existing
225     log "Found existing for #{mo_url}: #{existing}"
226     return existing
227   end
228   o = Observation.new( user: user, geoprivacy: Observation::OPEN )
229   o.observation_field_values.build( observation_field: mo_url_observation_field, value: mo_url )
230   if ( location = result.at_css( "> location" ) )
231     o.place_guess = location.at( "name" ).text
232     swlat = location.at( "latitude_south" ).text.to_f
233     swlng = location.at( "longitude_west" ).text.to_f
234     nelat = location.at( "latitude_north" ).text.to_f
235     nelng = location.at( "longitude_east" ).text.to_f
236     o.latitude = swlat + ( ( nelat - swlat ) / 2 )
237     o.longitude = swlng + ( ( nelng - swlng ) / 2 )
238     o.positional_accuracy = lat_lon_distance_in_meters( o.latitude, o.longitude, nelat, nelng )
239   end
240   if ( latitude = result.at( "latitude" ) ) && ( longitude = result.at( "longitude" ) )
241     o.latitude = latitude.text.to_f
242     o.longitude = longitude.text.to_f
243     o.positional_accuracy = nil
244     # nesting this here b/c we probably shouldn't set geoprivacy unless viewer
245     # has permission to see the true coordinates
246     if ( gps_hidden = result.at_css( "> gps_hidden" ) ) && gps_hidden[:value] == "true"
247       o.geoprivacy = Observation::PRIVATE
248     end
249   end
250   # This would prioritize the owner's name if we wanted to do that, but after
251   # implementing this, I feel like it's probably preferrable to use the
252   # consensus name
253   # owner_naming = result.search( "> namings / naming" ).detect{|n|
254   #   n.at( "owner" ).try(:[], :id) == mo_user_id
255   # }&.at( "name" )
256   owner_naming = nil
257   consensus_naming = result.at( "consensus_name" )
258   [owner_naming, consensus_naming].compact.each do | naming |
259     name = naming.at( "name" ).text
260     taxon = taxon_for_name( name )
261     if o.taxon.blank? && taxon && taxon.persisted?
262       o.taxon = taxon
263       if taxon.name != name
264         warn( mo_url, "Name mismatch, #{name} on MO, #{taxon.name} on iNat" )
265       end
266     end
267     o.species_guess ||= name
268   rescue ActiveRecord::AssociationTypeMismatch
269     warn( mo_url, "Failed to import a new taxon for #{name}" )
270   end
271   if consensus_naming
272     name = consensus_naming.at( "name" ).text
273     o.observation_field_values.build( observation_field: mo_name_observation_field, value: [
274       name,
275       consensus_naming.at( "author" ).try( :text )
276     ].compact.join( " " ) )
277   end
278   if owner_naming
279     name = owner_naming.at( "name" ).text
280     o.observation_field_values.build( observation_field: mo_name_observation_field, value: [
281       name,
282       owner_naming.at( "author" ).try( :text )
283     ].compact.join( " " ) )
284   end
285 
286   if ( date = result.at_css( "> date" ) )
287     o.observed_on_string = date.text
288   end
289   if ( notes = result.at_css( "> notes" ) )
290     if notes.text =~ /:Other/
291       o.description = notes.text[/&gt;&quot;(.+)&quot;\}/, 1]
292     elsif notes.text != "<p>{}</p>"
293       o.description = notes.text
294     end
295   end
296   if ( notes_fields = result.at_css( "> notes_fields" ) )
297     o.description ||= ""
298     notes_fields_hash = notes_fields.children.each_with_object( {} ) do | notes_field, memo |
299       key = notes_field.at( "key" ).try( :text )
300       val = notes_field.at( "value" ).try( :text )
301       memo[key] = val if key && val
302     end
303     unless notes_fields_hash.blank?
304       o.description += "\n\n#{notes_fields_hash.map {| k, v | "<strong>#{k}</strong>: #{v}" }.join( "\n\n" )}"
305     end
306   end
307   if !options[:skip_images] && ( images = images_from_result( result ) ) && images.size.positive?
308     images.each do | image |
309       image_url = image.at( :original_url ).text
310       lp = LocalPhoto.new( user: user )
311       begin
312         log "getting image from #{image_url}"
313         io = URI.parse( image_url ).open
314         Timeout.timeout( 10 ) do
315           lp.file = ( io.base_uri.path.split( "/" ).last.blank? ? nil : io )
316         end
317       rescue StandardError
318         warn( mo_url, "Failed to download #{image_url}" )
319         next
320       end
321       if ( image_license = image.at( "license" ) )
322         lp.license = case image_license[:url]
323         when %r{by-nc-sa/} then Photo::CC_BY_NC_SA
324         when %r{by-nc-nd/} then Photo::CC_BY_NC_ND
325         when %r{by-sa/} then Photo::CC_BY_SA
326         when %r{by-nd/} then Photo::CC_BY_ND
327         when %r{by/} then Photo::CC_BY
328         when %r{publicdomain/} then Photo::PD
329         end
330       end
331       o.observation_photos.build( photo: lp )
332     end
333   end
334   o
335 end

 69 def run
 70   return if Date.today >= REMOVAL_DATE
 71 
 72   update( finished_at: nil, error: nil, exception: nil )
 73   log "Importing observations from Mushroom Observer for #{user}"
 74   page = 1
 75   errors = {}
 76   loop do
 77     results = get_results_xml( page: page )
 78     break if results.blank?
 79 
 80     results.each do | result |
 81       transaction do
 82         o = observation_from_result( result )
 83         if o && !o.save
 84           mo_url = mo_url_from_result( result )
 85           errors[mo_url] = o.errors.full_messages.to_sentence
 86           clear_warnings_for_url( mo_url )
 87         end
 88       end
 89     end
 90     page += 1
 91   end
 92   log "Finished importing observations from Mushroom Observer for #{user}"
 93   Emailer.moimport_finished( self, errors, @warnings ).deliver_now
 94 rescue StandardError => e
 95   exception_string = [e.class, e.message].join( " :: " )
 96   log "Error: #{exception_string}\n#{e.backtrace}"
 97   update(
 98     finished_at: Time.now,
 99     error: "Error",
100     exception: user.is_admin? ? [exception_string, e.backtrace].join( "\n" ) : exception_string
101   )
102   Emailer.moimport_finished( self, errors, @warnings ).deliver_now
103   false
104 end
