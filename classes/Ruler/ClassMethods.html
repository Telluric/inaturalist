15 def validates_rules_from(association, options = {})
16   validation_method_name = "validate_rules_from_#{association.to_s}"
17   rule_methods = options.delete(:rule_methods) || []
18   define_method(validation_method_name) do
19     return if send(association).blank?
20     rules = send(association).send("#{self.class.to_s.underscore.singularize}_rules")
21     rules.group_by(&:operator).each do |operator, operator_rules|
22       errors_for_operator = []
23       operator_rules.each do |rule|
24         if rule.validates?(self)
25           # since only one of the group needs to pass, we can stop
26           break
27         else
28           errors_for_operator << rule.terms
29         end
30       end
31       next if errors_for_operator.blank?
32       if operator_rules.size == 1
33         errors.add( :base, "Didn't pass rule: #{errors_for_operator.first}" )
34       # FYI: if there are multiple rules with the same operator
35       # ONLY ONE of the rules with that operator must pass. For example
36       # if there are 10 place rules, the obs needs be in only 1
37       elsif errors_for_operator.size == operator_rules.size
38         errors.add( :base, "Didn't pass rules: #{errors_for_operator.join(' OR ')}" )
39       end
40     end
41   end
42   validate validation_method_name.to_sym, options
43   
44   const_set "RULE_METHODS", rule_methods
45 end
