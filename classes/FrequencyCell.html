69   def self.export( path )
70     sql = <<-SQL
71       SELECT
72         fc.swlat::varchar||'.'||fc.swlng::varchar as key,
73         fcmt.month,
74         fcmt.taxon_id,
75         fcmt.count,
76         CASE WHEN t.rank_level > 10 THEN 't' ELSE NULL END higher_taxon
77       FROM
78         frequency_cells fc
79           JOIN frequency_cell_month_taxa fcmt ON fc.id = fcmt.frequency_cell_id
80           JOIN taxa t ON fcmt.taxon_id = t.id
81       WHERE
82         t.rank_level = 10
83       ORDER BY
84         fc.swlat ASC,
85         fc.swlng ASC
86     SQL
87     self.export_sql_to_csv( sql, path )
88   end

 4 def self.populate
 5   con = ActiveRecord::Base.connection

 8   con.execute( "TRUNCATE TABLE frequency_cells RESTART IDENTITY" )

10     lat = -90
11     while lat <= ( 90 - CELL_SIZE )
12       lng = -180
13       lat_cells = []
14       while lng <= ( 180 - CELL_SIZE )
15         lat_cells << "(#{lat},#{lng})"
16         lng += CELL_SIZE
17       end
18       sql = "INSERT INTO frequency_cells (swlat, swlng) VALUES " + lat_cells.join( "," )
19       con.execute( sql )
20       lat += CELL_SIZE
21     end
22   end
23 
24   # iterate through the grid and populate with FrequencyCellMonthTaxa
25   con.execute( "TRUNCATE TABLE frequency_cell_month_taxa RESTART IDENTITY" )
26   start_time = Time.now
27   cells_counted = 0
28   Benchmark.measure do
29     lat = -90
30     while lat <= ( 90 - CELL_SIZE )
31       lng = -180
32       FrequencyCellMonthTaxon.transaction do
33         while lng <= ( 180 - CELL_SIZE )
34           params = {
35             swlat: lat,
36             swlng: lng,
37             nelat: lat + CELL_SIZE,
38             nelng: lng + CELL_SIZE,
39             quality_grade: "research,needs_id",
40             taxon_is_active: "true",
41             identifications: "most_agree"
42           }
43           response = INatAPIService.get( "/observations/taxa_counts_by_month",
44             params, { retry_delay: 2.0, retries: 30, json: true }
45           )
46           month_of_year = response && response["results"] && response["results"]["month_of_year"]
47           if month_of_year && !month_of_year.blank?
48             frequency_cell = FrequencyCell.where( swlat: lat, swlng: lng ).first
49             month_of_year.each do |month, leaf_counts|
50               sql = "INSERT INTO frequency_cell_month_taxa (frequency_cell_id, month, taxon_id, count) VALUES " +
51                 leaf_counts.filter{ |r| !r["taxon_id"].blank? }.map {|r| "(#{frequency_cell.id},#{month},#{r["taxon_id"]},#{r["count"]})" }.join( "," )
52               con.execute( sql )
53             end
54             puts "Inserted #{lat},#{lng} :: #{month_of_year.map{ |m,v| v.length }.inject(:+)}"
55           end
56           lng += CELL_SIZE
57           cells_counted += 1
58           total_time = Time.now - start_time
