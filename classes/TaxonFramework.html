175 def self.bud_off_child_taxon_framework( taxon_id )
176   return false unless taxon = Taxon.find(taxon_id)
177   return false unless parent_taxon_framework = taxon.get_upstream_taxon_framework
178   ancestry = [taxon.ancestor_ids, taxon_id].join( "/" )
179   tf = TaxonFramework.new(
180     taxon_id: taxon_id,
181     rank_level: parent_taxon_framework.rank_level,
182     source_id: parent_taxon_framework.source_id,
183     user_id: parent_taxon_framework.user_id
184   )
185   tf.skip_check_frameworks = true
186   return false unless tf.save
187   tfrs = TaxonFrameworkRelationship.joins( :taxa ).where( "taxon_framework_id = ? AND ( taxa.ancestry = '#{ancestry}' OR taxa.ancestry LIKE '#{ancestry}/%' )", parent_taxon_framework.id )
188   tfrs.update_all( taxon_framework_id: tf.id )
189   true
190 end

132     where( is_active: true ).
