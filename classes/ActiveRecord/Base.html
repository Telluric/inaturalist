158 def known_spam?
159   if self.is_a?(GuideTaxon)
160     # guide taxa can have many sections, each of which could be spam,
161     # so when one is spam consider the entire GuideTaxon as spam
162     if self.guide_sections.any?{ |s| s.flagged_as_spam? }
163       return true
164     end
165     return false
166   end
167   if (self.respond_to?(:flagged_as_spam?) && self.flagged_as_spam?)
168     return true
169   end
170   false
171 end
