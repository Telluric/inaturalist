 85 def self.data( options = {} )
 86   unless options[:core] == DarwinCore::Cores::TAXON
 87     raise "VernacularNames extension can only be used with a taxon core"
 88   end
 89   # fname = "vernacular_names.csv"
 90   if options[:work_path]
 91     work_path = options[:work_path]
 92   else
 93     work_path = Dir.mktmpdir
 94     FileUtils.mkdir_p work_path, mode: 0755
 95   end
 96   paths = []
 97   file_names.each do |lexicon, fname|
 98     tmp_path = File.join( work_path, fname )
 99     mode = File.exists?( tmp_path ) ? "a" : "w"
100     CSV.open( tmp_path, mode ) do |csv|
101       csv << TERM_NAMES if mode == "w"
102       DarwinCore::VernacularName.base_scope.
103           where( lexicon: lexicon ).
104           includes(:taxon, :source, :creator, :updater, place_taxon_names: :place).
105           order("taxon_id").
106           find_each do |tn|
107         DarwinCore::VernacularName.adapt( tn, core: options[:core] )
108         csv << DarwinCore::VernacularName::TERMS.map{|field, uri, default, method| tn.send(method || field)}
109       end
110     end
111     paths << tmp_path
112   end
113   paths
114 end
