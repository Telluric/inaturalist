83 def cached_ancestor(rank)
84   return nil if rank_level > ::Taxon.const_get("#{rank.to_s.upcase}_LEVEL")
85   cache = DarwinCore::Taxon.instance_variable_get("@#{rank}_cache")
86   t = nil
87   ancestor_ids.each do |aid|
88     break if t = cache[aid]
89   end
90   unless t
91     if ancestor = send("find_#{rank}")
92       t = ancestor.name
93       cache[ancestor.id] = t
94     end
95   end
96   t
97 end
