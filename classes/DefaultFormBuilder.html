161   options = field_content if block_given?
162   options ||= {}

164   description_after = options.delete( :description_after )

169

172     if options[:field_name] == "radio_button"

174     end
175     label_options = { class: options[:label_class] }
176     label_options[:for] = options[:id] if options[:id]
177     label_tag = label( label_field, options[:label].to_s.html_safe, label_options )
178     if options[:required]
179       label_tag += content_tag( :span, " *", class: "required" )
180     end
181     label_content = content_tag( label_after ? :span : :div, label_tag, class: "inlabel" )
182   end
183 
184   description = content_tag( :div, options[:description], class: "description" ) if options[:description]
185   content = "#{content}#{block_given? ? @template.capture( &block ) : field_content}"
186 
187   datalist = nil
188   if options[:datalist]
189     datalist = content_tag(
190       :datalist,
191       options[:datalist].map {| item | content_tag( :option, item, value: item ) }.join( "\n" ).html_safe,
192       id: dom_id( object, "#{field}-datalist" )
193     )
194   end
195 
196   content = if label_after
197     "#{content} #{datalist} #{label_content} #{description}"
198   elsif description_after
199     "#{label_content} #{content} #{datalist} #{description}"
200   else
201     "#{label_content} #{description} #{content} #{datalist}"
202   end
203 
204   @template.content_tag( :div, content.html_safe, wrapper_options )

107 def sort_lexicons( lexicons )
108   sortable_locale = I18N_LOCALES.detect {| l | l =~ /#{I18n.locale}-phonetic/ }
109   if sortable_locale
110     lexicons.sort do | a, b |
111       t_a = I18n.t(
112         "lexicons.#{a.gsub( ' ', '_' ).gsub( '-', '_' ).gsub( /[()]/, '' ).downcase}",
113         locale: sortable_locale,
114         default: ""
115       )
116       t_b = I18n.t(
117         "lexicons.#{b.gsub( ' ', '_' ).gsub( '-', '_' ).gsub( /[()]/, '' ).downcase}",
118         locale: sortable_locale,
119         default: ""
120       )
121       # Make sure sortable translations appear first
122       [( t_a == "" ? 1 : 0 ), t_a] <=> [( t_b == "" ? 1 : 0 ), t_b]
123     end
124   else
125     lexicons.sort_by do | lexicon |
126       key = "lexicons.#{lexicon.gsub( ' ', '_' ).gsub( '-', '_' ).gsub( /[()]/, '' ).downcase}"
127       I18n.t( key, default: lexicon ).to_s.downcase
128     end
129   end
130 end

 72 def time_zone_select( method, priority_zones = nil, options = {}, html_options = {} )
 73   options[:include_blank] = true if options[:include_blank].nil?
 74   html_options[:class] = "#{html_options[:class]} time_zone_select"
 75   zone_options = "".html_safe
 76   selected = options.delete( :selected )
 77   model = options.delete( :model ) || ActiveSupport::TimeZone
 78   zones = model.all
 79   convert_zones = lambda {| list, selected_zone |
 80     list.map do | z |
 81       opts = {
 82         :value => z.name,
 83         "data-time-zone-abbr" => z.tzinfo.current_period.abbreviation,
 84         "data-time-zone-tzname" => z.tzinfo.name,
 85         "data-time-zone-offset" => z.utc_offset,
 86         "data-time-zone-formatted-offset" => z.formatted_offset
 87       }
 88       opts[:selected] = "selected" if selected_zone == z.name
 89       content_tag( :option, z.to_s, opts )
 90     end.join( "\n" )
 91   }
 92   if priority_zones
 93     if priority_zones.is_a?( Regexp )
 94       priority_zones = model.all.grep( priority_zones )
 95     end
 96     zone_options += convert_zones.call( priority_zones, selected ).html_safe
 97     zone_options += "<option value=\"\" disabled=\"disabled\">-------------</option>\n".html_safe
 98     zones = zones.reject {| z | priority_zones.include?( z ) }
 99   end
100   zone_options += convert_zones.call( zones, selected ).html_safe
101   tag = INatInstanceTag.new(
102     object_name, method, self
103   ).to_select_tag_with_option_tags( zone_options, options, html_options )
104   form_field method, tag, options.merge( html_options )
105 end
