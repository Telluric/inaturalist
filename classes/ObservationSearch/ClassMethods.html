706 def elastic_user_taxon_counts(elastic_params, options = {})
707   options[:limit] ||= 500
708   aggregation_user_limit = 10000
709   elastic_params[:filters] ||= [ ]
710   elastic_params[:filters] << { range: {
711     "taxon.rank_level" => { lte: Taxon::RANK_LEVELS["species"] } } }
712   # We've started running into memory problems with ES not being able to
713   # handle some aggregates on a large scale. We will query for users in
714   # batches of 10,000, so if there are fewer than that we can query now.
715   if( options[:batch] == false ||
716       (options[:count_users] && options[:count_users] <= aggregation_user_limit) )
717     return elastic_user_taxon_counts_batch(elastic_params, options)
718   end
719   # fetch a list of every user_id whose observations match the search
720   user_counts = Observation.elastic_search(elastic_params.merge(size: 0, aggregate: {
721     user_observations: { "user.id": 100000 }
722   })).response.aggregations
723   user_ids = user_counts.user_observations.buckets.map{ |b| b["key"] }
724   counts = [ ]
725   # in batches, search ES with the original query filtered by the batch IDs
726   user_ids.each_slice(aggregation_user_limit) do |batch_user_ids|
727     filters = elastic_params[:filters] + [
728       { terms: { "user.id" => batch_user_ids } } ]
729     counts += elastic_user_taxon_counts_batch(elastic_params.merge(filters: filters), options)
730   end
731   # sort by count descending and return the top `limit`
732   counts.sort_by{ |b| b["count_all"] }.reverse[0...options[:limit]]
733 end

377   p[:user_id] = p[:user_id] || p[:user]

379   # Handle multiple users in user_id

381     candidate = user_id.to_s.strip
382     User.find_by_id( candidate ) || User.find_by_login( candidate )
383   end.compact
384   p[:user_id] = users.blank? ? nil : users.map(&:id)
385   p[:user_id] = p[:user_id].first if p[:user_id].is_a?( Array ) && p[:user_id].size == 1
386 
387   unless p[:user_id].blank? || p[:user_id].is_a?(Array)
388     p[:user] = User.find_by_id(p[:user_id])
389     p[:user] ||= User.find_by_login(p[:user_id])
390   end
391   if p[:user].blank? && !p[:login].blank?
392     p[:user] ||= User.find_by_login(p[:login])
393   end
394 
395   unless p[:ident_user_id].blank?
396     ident_users = p[:ident_user_id].is_a?( Array ) ?
397       p[:ident_user_id] : [p[:ident_user_id].to_s.split( "," )].flatten
398     ident_user_ids = []
399     ident_users.each do | id_or_login |
400       id_or_login = id_or_login.strip
401       ident_user = User.find_by_id( id_or_login )
402       ident_user ||= User.find_by_login( id_or_login )
403       ident_user_ids << ident_user.id if ident_user
404     end
405     p[:ident_user_id] = ident_user_ids.join( "," )
406   end
407 
408   unless p[:projects].blank?
409     project_ids = [p[:projects]].flatten
410     p[:projects] = Project.find(Project.slugs_to_ids(project_ids))
411     p[:projects] = p[:projects].compact
412     if p[:projects].blank?
413       project_ids.each do |project_id|
414         p[:projects] += Project.find(Project.slugs_to_ids(project_id))
415       end
416       p[:projects] = p[:projects].flatten.compact
417     end
418   end
419 
420   if p[:pcid] && p[:pcid] != 'any'
421     p[:pcid] = p[:pcid].yesish?
422   end
423 
424   unless p[:not_in_project].blank?
425     p[:not_in_project] = Project.find(p[:not_in_project]) rescue nil
426   end
427 
428   p[:rank] = p[:rank] if Taxon::VISIBLE_RANKS.include?(p[:rank])
429   p[:hrank] = p[:hrank] if Taxon::VISIBLE_RANKS.include?(p[:hrank])
430   p[:lrank] = p[:lrank] if Taxon::VISIBLE_RANKS.include?(p[:lrank])
431 
432   p.each do |k,v|
433     p[k] = nil if v.is_a?(String) && v.blank?
434   end
435 
436   p[:_query_params_set] = true
437   p
438 end
