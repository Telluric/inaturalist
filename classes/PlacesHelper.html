60 def google_charts_map_for_places(places, options = {}, tag_options = {})
61   countries = places.select {|p| p.place_type == Place::PLACE_TYPE_CODES['Country']}
62   states = places.select {|p| p.admin_level == Place::STATE_LEVEL && p.parent.try(:code) == 'US'}
63   geographical_area = 'world'
64   labels = countries.map(&:code)
65   if states.size > 0 && (countries.empty? || (countries.size == 1 && countries.first.code = 'US'))
66     labels = states.map {|s| s.code.gsub('US-', '')}
67     geographical_area = 'usa'
68   end
69   data = "t:#{(['100'] * labels.size).join(',')}"
70   url_for_options = {
71     :host => 'chart.googleapis.com',
72     :controller => 'chart',
73     :port => nil,
74     :chs => '440x220',
75     :chco => 'EEEEEE,1E90FF,1E90FF',
76     :chld => labels.join,
77     :chd => labels.empty? ? 's:_' : data,
78     :cht => 't',
79     :chtm => geographical_area
80   }.merge(options)
81   
82   tag_options[:alt] = "Map of #{geographical_area == 'usa' ? 'US states' : 'countries'}: #{labels.join(', ')}"
83   tag_options[:width] ||= url_for_options[:chs].split('x').first
84   tag_options[:height] ||= url_for_options[:chs].split('x').last
85   
86   image_tag(
87     url_for(url_for_options),
88     tag_options
89   )
90 end

51 def map_zoom_for_place(place, width, height)
52   return 0 if [place.nelat, place.nelng, place.swlat, place.swlng].include?(nil)
53   (1..SPHERICAL_MERCATOR.levels).to_a.reverse_each do |zoom_level|
54     minx, miny = SPHERICAL_MERCATOR.from_ll_to_pixel([place.swlng, place.swlat], zoom_level)
55     maxx, maxy = SPHERICAL_MERCATOR.from_ll_to_pixel([place.nelng, place.nelat], zoom_level)
56     return zoom_level if (maxx - minx < width) && (maxy - miny < height)
57   end
58 end

 92 def nested_place_list(*args, &block)
 93   arranged = if args.first.is_a?(ActiveSupport::OrderedHash)
 94     args.first
 95   else
 96     candidates = args.first.is_a?(Array) ? args.first.compact.flatten : args.compact.flatten
 97     places = candidates if candidates.first.is_a?(Place)
 98     places ||= candidates.map(&:place).compact
 99     Place.arrange_nodes(places.sort_by{|p| p.ancestry.to_s + "#/0/{#{p.name}}"})
100   end
101   content_tag :ul do
102     arranged.map do |place, children|
103       li = if block_given?
104         capture(place, &block)
105       else
106         link_to(place.display_name, place)
107       end
108       li += nested_place_list(children, &block) unless children.blank?
109       content_tag :li, li
110     end.join(' ').html_safe
111   end.html_safe
