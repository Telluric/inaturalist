161 def self.new_from_eol_data_object(data_object)
162   gs = GuideSection.new(
163     :title => (data_object.at('title') || data_object.at('subject')).content.split('#').last.underscore.humanize,
164     :description => data_object.at('description').try(:content),
165     :rights_holder => data_object.at('rightsHolder').try(:content) || data_object.at('agent[role=compiler]').try(:content)
166   )
167   gs.license = case data_object.at('license').to_s
168   when /\/by\// then Observation::CC_BY
169   when /\/by-nc\// then Observation::CC_BY_NC
170   when /\/by-sa\// then Observation::CC_BY_SA
171   when /\/by-nd\// then Observation::CC_BY_ND
172   when /\/by-nc-sa\// then Observation::CC_BY_NC_SA
173   when /\/by-nc-nd\// then Observation::CC_BY_NC_ND
174   when /\/publicdomain\// then Photo::PD_CODE
175   else
176     data_object.at('license').to_s
177   end
178   gs.source_url = if (verson_id = data_object.at('dataObjectVersionID').try(:content))
179     "http://eol.org/data_objects/#{verson_id}"
180   elsif data_object.at('source').to_s =~ /http/
181     data_object.at('source').to_s
182   end
183   gs
184 end
