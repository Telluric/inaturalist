339 def self.add_new_listed_taxa(taxon, new_place_ids, current_place_ids, taxon_ids)
340   check_lists = CheckList.where(place_id: new_place_ids).
341     joins("JOIN places ON places.check_list_id = lists.id")
342   atlases = Atlas.where( "is_active = ? AND taxon_id IN ( ? )", true, taxon_ids )
343   complete_sets = CompleteSet.where(
344     "is_active =  ? AND taxon_id IN ( ? ) AND place_id IN ( ? )",
345     true, taxon_ids, current_place_ids
346   )
347   check_lists.each do |list|
348     if [Place::COUNTRY_LEVEL ,Place::STATE_LEVEL ,Place::COUNTY_LEVEL].include?( list.place.admin_level )
349       if atlases.exists?
350         atlas_that_excludes_check_list_place = atlases.detect do |atlas|
351           # If an atlas's presence places do not overlap with the list's place
352           # and ancestor places, the atlas excludes the list's place
353           ( atlas.presence_places.pluck(:id) & list.place.self_and_ancestor_ids ).size == 0
354         end
355         if atlas_that_excludes_check_list_place
356           next
357         end
358       end
359       if complete_sets.exists?
360         complete_set_that_excludes_taxon = complete_sets.detect do |complete_set|
361           # A complete set implies that there are listed taxa for all species
362           # in the taxon for that place, so if there is no listed taxon for
363           # this taxon in the complete set place, it should be excluded
364           !ListedTaxon.where( place_id: complete_set.place_id, taxon_id: taxon.id ).exists?
365         end
366         if complete_set_that_excludes_taxon
367           next
368         end
369       end
370     end
371     list.add_taxon( taxon, force_update_cache_columns: true )
372     if taxon.rank_level < Taxon::SPECIES_LEVEL
373       list.add_taxon( taxon.species, force_update_cache_columns: true )
374     end
375   end
376 end

253 def self.refresh_with_observation(observation, options = {})
254   observation, observation_id = CheckList.get_observation_to_refresh(observation)
255   options[:observation_id] ||= observation_id
256   taxon_ids = CheckList.get_taxon_ids_to_refresh(observation, options)
257   return if taxon_ids.blank?
258   Rails.logger.info "[INFO #{Time.now}] refresh_with_observation #{observation_id}, taxon_ids: #{taxon_ids.inspect}"
259   
260   current_place_ids = CheckList.get_current_place_ids_to_refresh(observation, options)
261   Rails.logger.info "[INFO #{Time.now}] refresh_with_observation #{observation_id}, current_place_ids: #{current_place_ids.inspect}"
262   
263   current_listed_taxa = ListedTaxon.where(place_id: current_place_ids, taxon_id: taxon_ids)
264   current_listed_taxa_of_this_taxon = current_listed_taxa.select{|lt| lt.taxon_id == observation.taxon_id}
265   new_place_ids = current_place_ids - current_listed_taxa_of_this_taxon.map{|lt| lt.place_id}
266   Rails.logger.info "[INFO #{Time.now}] refresh_with_observation #{observation_id}, new_place_ids: #{new_place_ids.inspect}"
267   
268   old_place_ids = CheckList.get_old_place_ids_to_refresh(observation, options)
269   Rails.logger.info "[INFO #{Time.now}] refresh_with_observation #{observation_id}, old_place_ids: #{old_place_ids.inspect}"
270   
271   old_listed_taxa = ListedTaxon.where(place_id: (old_place_ids - current_place_ids), taxon_id: taxon_ids)
272   listed_taxa = (current_listed_taxa + old_listed_taxa).compact.uniq
273   unless listed_taxa.blank? || options[:new]
274     Rails.logger.info "[INFO #{Time.now}] refresh_with_observation #{observation_id}, updating #{listed_taxa.size} existing listed taxa"
275     listed_taxa.each do |lt|
276       CheckList.delay(priority: INTEGRITY_PRIORITY, queue: "slow", run_at: 2.hours.from_now,
277         unique_hash: { "CheckList::refresh_listed_taxon": lt.id }
278       ).refresh_listed_taxon( lt.id )
279     end
280   end
281   if observation && observation.research_grade? && observation.taxon.species_or_lower?
282     Rails.logger.info "[INFO #{Time.now}] refresh_with_observation #{observation_id}, adding new listed taxa"
283     add_new_listed_taxa( observation.taxon, new_place_ids, current_place_ids, taxon_ids )
284   end
285   Rails.logger.info "[INFO #{Time.now}] refresh_with_observation #{observation_id}, finished"
286 end

170 def refresh(options = {})
171   finder = ListedTaxon.all
172   if taxa = options[:taxa]
173     finder = finder.where(list_id: self.id, taxon_id: taxa)
174   else
175     finder = finder.where(list_id: self.id)
176   end
177   
178   finder.includes({
179     taxon: [:taxon_names],
180     list: [:rules],
181     last_observation: [:taxon] }).find_in_batches do |batch|
182     batch.each do |listed_taxon|
183       listed_taxon.skip_update_cache_columns = options[:skip_update_cache_columns]
184       listed_taxon.skip_index_taxon = true
185       # re-apply list rules to the listed taxa
186       listed_taxon.force_update_cache_columns = true
187       listed_taxon.check_primary_listing
188       listed_taxon.save
189       # make sure we don't force update yet again when just validating
190       listed_taxon.force_update_cache_columns = false
191       if !listed_taxon.valid?
192         Rails.logger.debug "[DEBUG] #{listed_taxon} wasn't valid, so it's being " +
193           "destroyed: #{listed_taxon.errors.full_messages.join(', ')}"
194         listed_taxon.destroy
195       elsif listed_taxon.auto_removable_from_check_list?
196         listed_taxon.destroy
197       end
198     end
199     batch.map( &:taxon_id ).uniq.each do |taxon_id|
200       Taxon.delay(priority: INTEGRITY_PRIORITY, run_at: 1.hour.from_now,
201         unique_hash: { "Taxon::elastic_index": taxon_id }).
202         elastic_index!(ids: [taxon_id])
203     end
204   end
205   true
206 end
