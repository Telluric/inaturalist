 79 def self.from_observation_params(params, fieldset_index, owner)
 80   sounds = []
 81   unless Rails.env.production?
 82     SoundcloudSound
 83     LocalSound
 84   end
 85   (self.subclasses || []).each do |klass|
 86     klass_key = klass.to_s.underscore.pluralize.to_sym
 87     if params[klass_key] && params[klass_key][fieldset_index.to_s]
 88       if klass == SoundcloudSound
 89         params[klass_key][fieldset_index.to_s].each do |sid|
 90           sound = klass.new_from_native_sound_id(sid, owner)
 91           sound.user = owner
 92           sound.native_realname = owner.soundcloud_identity.native_realname
 93           sounds << sound
 94         end
 95       else
 96         params[klass_key][fieldset_index.to_s].each do |file_or_id|
 97           sound = if file_or_id.is_a?( ActionDispatch::Http::UploadedFile )
 98             LocalSound.new( file: file_or_id )
 99           else
100             Sound.find_by_id( file_or_id )
101           end
102           next unless sound
103           sound.user = owner
104           sound.native_realname = owner.name
105           sounds << sound
106         end
107       end
108     end
109   end
110   sounds
111 end
