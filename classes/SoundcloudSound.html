64 def to_taxa(options = {})
65   return [] unless self.native_response && (tags_string = self.native_response["tag_list"]).present?
66 
67   # convert tag_string to an array of strings and handle multiword tags (which soundcloud puts in quotes)
68   multiword_tags = tags_string.scan(/"[^"]+"/)
69   multiword_tags.map do |t| 
70     tags_string.gsub!(t, '') #remove from original set
71     t.gsub!("\"",'') #clean up the quotes
72   end
73   tags = tags_string.split(' ') + multiword_tags
74 
75   tags << self.native_response["title"]
76 
77   # First try to find taxa matching taxonomic machine tags, then default
78   # to all tags
79   machine_tags = tags.select{|t| t =~ /taxonomy\:/}
80   taxa = Taxon.tags_to_taxa(machine_tags, options) unless machine_tags.blank?
81   taxa ||= Taxon.tags_to_taxa(tags, options)
82   taxa
83   taxa.compact
84 end
