57 def year
58   @year = params[:year].to_i
59   if @year > Date.today.year || @year < 1950
60     return render_404
61   end
62   @display_user = User.find_by_login( params[:login] )
63   if !params[:login].blank? && !@display_user
64     return render_404
65   end
66   if @display_user && !current_user && !@display_user.locale.blank?
67     I18n.locale = @display_user.locale
68   end
69   @year_statistic = if @display_user
70     YearStatistic.where( "user_id = ? AND year = ?", @display_user, @year ).first
71   elsif Site.default.try(:id) == @site.id
72     YearStatistic.where( "user_id IS NULL and site_id IS NULL and year = ?", @year ).first
73   else
74     YearStatistic.where( site_id: @site, year: @year ).where( "user_id IS NULL" ).first
75   end
76   @headless = @footless = true
77   @shareable_image_url = if @year_statistic && @year_statistic.shareable_image?
78     @year_statistic.shareable_image
79   elsif @display_user && @display_user.icon?
80     @display_user.icon.url(:large)
81   elsif @site.shareable_image?
82     @site.shareable_image.url
83   end
84   respond_to do |format|
85     format.html do
86       @responsive = true
87       render layout: "bootstrap"
88     end
89     format.json {
90       if @year_statistic.blank?
91         render_404
92         return
93       end
94       render json: @year_statistic.data
95     }
96   end
97 end
