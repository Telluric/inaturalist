276     names = names.split( "\n" ).compact.map( &:strip )

297     end

132 def set_taxon
133   ancestry_counts = Taxon.joins( :guide_taxa ).where( "guide_taxa.guide_id = ?", id ).group( :ancestry ).count
134   ancestries = ancestry_counts.map {| a, _c | a.to_s.split( "/" ) }.sort_by( &:size ).compact
135   if ancestries.blank?
136     Guide.where( id: id ).update_all( taxon_id: nil )
137     return
138   end
139 
140   width = ancestries.last.size
141   matrix = ancestries.map do | a |
142     a + ( [nil] * ( width - a.size ) )
143   end
144 
145   # start at the right col (lowest rank), look for the first occurrence of
146   # consensus within a rank
147   consensus_taxon_id = nil
148   width.downto( 0 ) do | c |
149     column_taxon_ids = matrix.map {| ancestry | ancestry[c] }
150     if column_taxon_ids.uniq.size == 1 && !column_taxon_ids.first.blank?
151       consensus_taxon_id = column_taxon_ids.first
152       break
153     end
154   end
155 
156   Guide.where( id: id ).update_all( taxon_id: consensus_taxon_id )
157 end

398 def to_ngz( options = {} )
399   start_log_timer "#{self} to_ngz"
400   ordered_guide_taxa = guide_taxa.
401     order( :position ).
402     includes( { guide_photos: [:photo] }, :guide_ranges, :guide_sections, :tags )
403   image_sizes = %w(thumb small medium)
404   local_asset_path = "files"
405   work_path = ngz_work_path
406   FileUtils.mkdir_p work_path, mode: 0o755
407   # build xml and write to file in tmpdir
408   xml_fname = "#{id}.xml"
409   xml_path = File.join( work_path, xml_fname )
410   File.open xml_path, "w" do | f |
411     f.write FakeView.render( template: "guides/show", formats: [:xml], locals: {
412       local_asset_path: local_asset_path,
413       guide: self,
414       guide_taxa: ordered_guide_taxa,
415       image_sizes: image_sizes
416     } )
417   end
418 
419   # mkdir for assests
420   full_asset_path = File.join( work_path, local_asset_path )
421   FileUtils.mkdir_p full_asset_path, mode: 0o755
422   # loop over all photos and ranges, downloading assets to the asset dir
423   ordered_guide_taxa.each do | gt |
424     ( gt.guide_photos + gt.guide_ranges ).each do | gp |
425       image_sizes.each do | s |
426         next unless ( url = gp.send( "#{s}_url" ) )
427 
428         fname = FakeView.guide_asset_filename( gp, size: s )
429         path = File.join( full_asset_path, fname )
430         Rails.logger.info "[INFO #{Time.now}] Fetching #{url} to #{path}"
431         begin
432           File.open( path, "wb" ) do | f |
433             URI.parse( CGI.escape( url ) ).open do | fr |
434               f.write( fr.read )
435             end
436           end
437         rescue StandardError => e
438           Rails.logger.error "[ERROR #{Time.now}] Failed to download #{url}: #{e}"
439           next
440         end
441       end
442     end
443   end
444 
445   # zip up the results & return the path
446   zip_path = "#{work_path}.ngz"
447   system "cd #{work_path} && zip -qr #{zip_path} #{xml_fname} #{local_asset_path}"
448   Rails.logger.debug "[DEBUG] options[:path]: #{options[:path]}"
449   if options[:path]
450     Rails.logger.debug "[DEBUG] moving from #{zip_path} to #{options[:path]}"
451     FileUtils.mv( zip_path, options[:path] )
452   end
453   FileUtils.rm_rf work_path # clean up all those big files
454   end_log_timer
455   zip_path
456 end
