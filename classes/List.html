Class List < ApplicationRecord

  • app/models/list.rb

A List is a list of taxa. Naturalists often keep lists of taxa, whether they be lists of things they've seen, lists of things they'd like to see, or just lists of taxa that interest them for some reason.

 95 def generate_csv(options = {})
 96   controller = options[:controller] || FakeView.new
 97   attrs = %w(taxon_name description occurrence_status establishment_means adding_user_login first_observation 
 98      last_observation url created_at updated_at taxon_common_name)
 99   ranks = Taxon::RANK_LEVELS.select{|r,l| (Taxon::COMPLEX_LEVEL..Taxon::KINGDOM_LEVEL).include?(l) }.keys - [Taxon::GENUSHYBRID]
100   headers = options[:taxonomic] ? ranks + attrs : attrs
101   fname = options[:fname] || "#{to_param}.csv"
102   fpath = options[:path] || File.join(options[:dir] || Dir::tmpdir, fname)
103   
104   # Always generate file to tmp path first
105   tmp_path = File.join(Dir::tmpdir, fname)
106   FileUtils.mkdir_p File.dirname(tmp_path), :mode => 0755
107   
108   is_default_checklist = (is_a?(CheckList) && is_default?)
109   scope = if is_default_checklist
110     ListedTaxon.joins(:taxon).where(place_id: place_id)
111   else
112     ListedTaxon.where(list_id: id)
113   end
114   scope = scope.includes({ taxon: { taxon_names: :place_taxon_names } },
115     :user, :first_observation, :last_observation)
116   
117   ancestor_cache = {}
118   taxa_recorded = {}
119   CSV.open(tmp_path, 'w') do |csv|
120     csv << headers
121     scope.find_each do |lt|
122       next if lt.taxon.blank?
123       next if is_default_checklist && taxa_recorded[lt.taxon.id]
124       taxa_recorded[lt.taxon.id] = true if is_default_checklist
125       row = []
126       if options[:taxonomic]
127         ancestor_ids = lt.taxon.ancestor_ids.map{|tid| tid.to_i}
128         uncached_ancestor_ids = ancestor_ids - ancestor_cache.keys
129         if uncached_ancestor_ids.size > 0
130           Taxon.where(id: uncached_ancestor_ids).select(:id, :name, :rank).each do |t|
131             ancestor_cache[t.id] = t
132           end
133         end
134         ancestors = ancestor_ids.map{|tid| t = ancestor_cache[tid]; t.try(:name) == 'Life' ? nil : t}.compact
135         ancestors << lt.taxon
136         row += ranks.map do |rank|
137           ancestors.detect{|t| t.rank == rank}.try(:name)
138         end
139       end
140       attrs.each do |h|
141         row << case h
142         when 'adding_user_login'
143           lt.user_login
144         when 'url'
145           controller.instance_eval { listed_taxon_url(lt) }
146         when 'first_observation', 'last_observation' 
147           controller.instance_eval { observation_url(lt.send(h)) } if lt.send(h)
148         else
149           lt.send(h)
150         end
151       end
152       csv << row
153     end
154     csv << []
155     csv << ["List created at", created_at]
156     csv << ["List updated at", updated_at]
157     csv << ["List updated by", user.login] if user
158     csv << ["CSV generated at", Time.now.utc]
159   end
160   
161   # When the full file is ready, then move it over to the real path
162   begin
163     FileUtils.mkdir_p File.dirname(fpath), :mode => 0755
164     if tmp_path != fpath
165       FileUtils.mv tmp_path, fpath
166     end
167   rescue Errno::EINVAL => e
168     # chown fails for some reason, maybe NFS related
169     if e.message =~ /Invalid argument/
170       return nil
171     else
172       raise e
173     end
174   end
175   fpath
176 end

31 def rank_rule=(new_rank_rule)
32   return if rank_rule == new_rank_rule
33   return if rank_rule.blank?
34   rules.each do |rule|
