32 def photo_fields
33   @context = params[:context] || 'user'
34   if current_user.facebook_identity.blank?
35     return authorize
36   elsif current_user.facebook_api.blank?
37     @reauthorization_needed = true
38     return authorize
39   end
40     
41   if @context == 'groups'
42     @groups = facebook_groups(current_user)
43     render :partial => 'facebook/groups' and return
44   elsif @context == 'friends'
45     @friend_id = params[:object_id]
46     @friend_id = nil if @friend_id == 'null' || @friend_id.blank?
47     if @friend_id.blank?  # if context is friends, but no friend id specified, we want to show the friend selector
48       @friends = facebook_friends(current_user)
49       render :partial => 'facebook/friends' and return
50     else
51       @albums = facebook_albums(current_user, @friend_id)
52       friend_data = current_user.facebook_api.get_object(@friend_id)
53       @friend_name = friend_data['first_name']
54       render :partial => 'facebook/albums'
55     end
56   else # assume @context == 'user'
57     @albums = facebook_albums(current_user)
58     render :partial => 'facebook/albums' and return
59   end
60 rescue Koala::Facebook::APIError => e
61   raise e unless e.message =~ /OAuthException/
62   @reauthorization_needed = true
63   authorize
64 end
