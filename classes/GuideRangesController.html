107 def import_from_eol
108   eol = EolService.new(:timeout => 30)
109   eol_page_id = params[:eol_page_id]
110   if eol_page_id.blank?
111     pages = eol.search(params[:q], :exact => true)
112     eol_page_id = pages.at('entry/id').try(:content)
113   end
114   return if eol_page_id.blank?
115   page = eol.page(eol_page_id, :texts => 0, :images => 0, :sounds => 0, :videos => 0, :maps => 50, :subjects => "all", :details => true)
116   page.remove_namespaces!
117   page.search('dataObject').map {|data_object|
118     Rails.logger.debug "[DEBUG] data_object: #{data_object}"
119     gr = GuideRange.new_from_eol_data_object(data_object)
120     # gr.valid? ? gr : nil
121     gr
122   }.compact
123 rescue Timeout::Error => e
124   []
125 end
