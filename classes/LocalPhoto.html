    # File app/models/local_photo.rb
247 def reset_file_from_original
248   interpolated_original_url = FakeView.image_url( self.file.url(:original) )
249   
250   # If we're using local file storage and using some kind of development-ish
251   # setup, it probably means we're running a single server process, which
252   # means running a HEAD request while *this* request is running is going to
253   # cause problems, but it also means that the original file is *probably*
254   # there so we can skip this file reset business.
255   return if interpolated_original_url =~ /localhost/
256 
257   # If the original file is there under the current path, no need to do anything
258   return if Photo.valid_remote_photo_url?( interpolated_original_url )
259   
260   # If it's not, check the old path
261   attachment_opts = LocalPhoto.attachment_definitions[:file]
262   image_url_opts = if CONFIG.usingS3
263     { base_url: "#{attachment_opts[:s3_protocol]}://#{attachment_opts[:bucket]}" }
264   else
265     {}
266   end
267   old_interpolated_original_url = FakeView.image_url(
268     Paperclip::Interpolations.interpolate( "photos/:id/:style.:extension", self.file, "original" ),
269     image_url_opts
270   )
271   url = if Photo.valid_remote_photo_url?( old_interpolated_original_url )
272     old_interpolated_original_url
273   
274   # If it's not at the old path, use the cached original_url if it's not copyright infringement
275   elsif original_url !~ /copyright/
276     FakeView.image_url( original_url )
277 
278   # If this is a copyright violation AND we don't have access to an original file, we're screwed.
279   else
280     raise Photo::MissingPhotoError.new( "We no longer have access to the original file" )
281   end
282 
283   io = open( URI.parse( url ) )
284   Timeout::timeout(10) do
285     self.file = (io.base_uri.path.split('/').last.blank? ? nil : io)
286   end
287 end

    # File app/models/local_photo.rb
394 def to_tags(options = {})
395   tags = []
396   if !metadata.blank? && !metadata[:dc].blank?
397     tags += [metadata[:dc][:subject]].flatten.reject(&:blank?).map(&:strip)
398     if options[:with_title] && !metadata[:dc][:title].blank?
399       tags += [metadata[:dc][:title]].flatten.reject(&:blank?).map(&:strip)
400     end
401   end
402   if options[:with_file_name] && file.file? && !file.original_filename.blank?
403     # every string of letters except underscores (NOT not-words, underscores,
404     # or numbers). Ignore the last since it's almost always a file extension
405     # Note that I'm trying to handle unicode characters here, but there's
406     # still a high chance of encoding weirdness happening
407     words = file.original_filename.scan(/[\p{L}\p{M}\'\’]+/)
408     words = words.reject do |word|
409       word.size < 4 || word =~ /^original|img|inat|dsc.?|jpe?g|png|gif|open-uri$/i
410     end
411     # Collect all combinations of these words from 1-word combinations up to
412     # the combination that includes all words. Note that a combination
413     # preserves order, unlike permutations
414     tags += ( 1..words.size ).map {|i|
415       words.combination( i ).to_a.map{|c| c.join( " " ) }
416     }.flatten
417   end
418   tags
419 end
