   # File lib/wikipedia_service.rb
42 def summary_from_parsed( parsed )
43   hxml = Nokogiri::HTML(HTMLEntities.new.decode(parsed.at( "text" ).try( :inner_text )))
44   hxml.search('table').remove
45   hxml.search("//comment()").remove
46   # Remove all elements that aren't displayed
47   hxml.search( "//*[contains(@style,'display:none')]/text()" ).remove
48   summary = ( hxml.search("//p").detect{|node| !node.inner_html.strip.blank?} || hxml ).inner_html.to_s.strip
49   summary = sanitizer.sanitize(summary, :tags => %w(p i em b strong))
50   summary.gsub! /\[.*?\]/, ''
51   summary
52 end
