43 def confirm
44   # If the user is signed in, load the data from the UserParent record if they created it
45   if current_user
46     if @user_parent = UserParent.where( id: params[:id], parent_user_id: current_user ).first
47       @user_parent_name = @user_parent.name
48       @user_parent_email = @user_parent.email
49     end
50   else
51     @user_parent = UserParent.find_by_id( params[:id] )
52   end
53   unless @user_parent
54     respond_to do |format|
55       format.html { render_404 }
56     end
57     return
58   end
59   # Otherwise load it from the session so other people can't see the data in UserParent record
60   @user_parent_name ||= session[:user_parent_name] || ""
61   @user_parent_email ||= session[:user_parent_email] || ""
62   if @user_parent_name.blank? && @user_parent_email.blank?
63     respond_to do |format|
64       format.html { redirect_back_or_default new_user_parent_path }
65     end
66     return
67   end
68   respond_to do |format|
69     format.html
70   end
71 end
