 22       @resolved = @flags.select {|f| f.resolved }
 23       @moderator_actions = if @object.respond_to?(:moderator_actions)
 24         @object.moderator_actions.limit( 100 ).to_a
 25       end
 26       respond_to do |format|
 27         format.html { render layout: "bootstrap" }
 28       end
 29       return
 30     end
 31   end
 32   # a real index of all flags, which can be filtered by flag type
 33   @flag_type = params[:flag].to_s.tr("_", " ")
 34   @flag_types = [params[:flags]].flatten.compact
 35   @flag_types = ["other", Flag::INAPPROPRIATE] if @flag_types.blank?
 36   @resolved = if params[:resolved].blank?
 37     "no"
 38   elsif params[:resolved] == "any"
 39     "any"
 40   elsif params[:resolved].yesish?
 41     "yes"
 42   elsif params[:resolved].noish?
 43     "no"
 44   end
 45   @deleted = if params[:deleted].blank?
 46     "any"
 47   elsif params[:deleted] == "any"
 48     "any"
 49   elsif params[:deleted].yesish?
 50     "yes"
 51   elsif params[:deleted].noish?
 52     "no"
 53   end
 54   @flaggable_type = params[:flaggable_type] if Flag::TYPES.include?( params[:flaggable_type] )
 55   @user = User.where( login: params[:user_id] ).first || User.where( id: params[:user_id] ).first
 56   @user ||= User.where( login: params[:user_name] ).first
 57   @resolver = User.where( login: params[:resolver_id] ).first || User.where( id: params[:resolver_id] ).first
 58   @resolver ||= User.where( login: params[:resolver_name] ).first
 59   @flagger_type = params[:flagger_type] if %w(any auto user).include?( params[:flagger_type] )
 60   @flagger_type ||= "any"
 61   @taxon = Taxon.find_by_id( params[:taxon_id] )
 62   @flags = Flag.order( "created_at desc" )
 63   @flags = @flags.where( flaggable_user_id: @user.id ) if @user
 64   unless @flag_type.blank?
 65     @flags = @flags.where(flag: @flag_type)
 66   end
 67   @flags = @flags.where( flaggable_type: @flaggable_type ) unless @flaggable_type.blank?
 68   if @flag_types.include?( "other" )
