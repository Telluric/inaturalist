10 def describe( taxon )
11   title = nil
12   page_title = title
13   content = nil
14   if Rails.env.production?
15     # Redirects from the iNat taxon ID in Wikidata to Wikipedia in a
16     # particular locale. We're using it to grab the Wikipedia page title.
17     # Note that this will only work with the production database, since
18     # those are the taxon IDs Wikidata has ingested, so unfortunately this
19     # can't really be tested
20     lang = @locale.to_s.split( "-" ).first
21     if wikipedia_url = wikidata_wikipedia_url_for_taxon( taxon )
22       title = wikipedia_url.to_s.split( "/" ).last
23       # If we got a title from Wikidata, try to retrieve content for that
24       unless title.blank?
25         title = CGI.unescape( title )
26         page_title, content = content_for_title( title )
27         @page_urls[taxon.id] = wikipedia_url unless content.blank?
28       end
29     end
30   end
31   # If we didn't get any content from a Wikidata-specified Wikipedia title,
32   # fall back to using the taxon data
33   if content.blank?
34     title = taxon.wikipedia_title if title.blank?
35     title = taxon.name if title.blank?
36     page_title, content = content_for_title( title )
37   end
38   return if content.blank?
39   content + "<p class='inat-wikipedia-attribution'>#{I18n.t(:wikipedia_attribution_cc_by_sa_3_html, url: page_url( taxon ), title: page_title )}</p>"
40 end
