    # File lib/ratatosk/lib/ratatosk/name_providers/nzor.rb
137 def get_taxon
138   if is_sciname? and is_valid?
139     taxon = NZORTaxonAdapter.new(@hxml)
140   else
141     taxon = NZORTaxonAdapter.new(accepted_name_hxml)
142   end
143   # This is necessary because calling save here runs the validations,
144   # sees that the Taxon is new and declares the lexion validation ok,
145   # then saves the new taxon, which would fire the after save callback
146   # creating another taxon name, and then this taxon name gets
147   # created, resulting in duplicate, invalid taxon names.
148   taxon.skip_new_taxon_name = true
149 
150   taxon
151 end
