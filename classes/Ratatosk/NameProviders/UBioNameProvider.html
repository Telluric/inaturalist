42 def find(name)
43   begin
44     results = get_keepers(name, @service.simple_namebank_search(name))
45     rdfs = results.map do |r|
46       @service.lsid(:namespace => 'namebank', :object => r[:namebankID])
47     end.compact
48   rescue UBioConnectionError => e
49     raise NameProviderError, e.message
50   end
51   
52   taxon_names = rdfs.map do |rdf|
53     begin
54       UBioTaxonNameAdapter.new(rdf, :np => self)
55     rescue TaxonNameAdapterError => e
56       # We could also forget these errors and rely on the new TaxonNames
57       # being invalid. Not sure which way is best. KMU 2008-08
58       nil
59     end
60   end.compact
61   
62   # For synonyms in the same taxonomic group, only keep one (canonical
63   # if possible)
64   taxon_names_by_tgroup = taxon_names.group_by do |tn|
65     tgroup = tn.hxml.at('//ubio:taxonomicGroup').inner_text.strip rescue nil
66     tgroup
67   end
68   keepers = taxon_names_by_tgroup.delete(nil) || []
69   taxon_names_by_tgroup.each do |tgroup, tnames|
70     tnames.group_by(&:name).each do |tname, synonyms|
71       keeper = synonyms.detect do |s| 
72         (s.hxml.at('//ubio:lexicalStatus').inner_text rescue nil) == 'Canonical form'
73       end
74       keeper ||= synonyms.first
75       keepers << keeper
76     end
77   end
78   
79   # Try to sort the names so canonicals are first
80   keepers = keepers.sort do |a,b|
81     a_canonical = (a.hxml.at('//ubio:lexicalStatus').inner_text rescue nil) == 'Canonical form'
82     b_canonical = (b.hxml.at('//ubio:lexicalStatus').inner_text rescue nil) == 'Canonical form'
83     if a_canonical && !b_canonical
84       -1
85     elsif b_canonical && !a_canonical
86       1
87     else
88       0
89     end
90   end
91   
92   keepers
93 end

 95 def get_lineage_for(taxon)
 96   # search cbank for this taxon in its many classifications
 97   cbankr_results = @service.classificationbank_search(:namebankID => taxon.source_identifier)
 98   
 99   # choose a classification, preferrably a nice and shiny one
100   cbank_id = choose_cbank_id(cbankr_results)
101 
102   # call uBio again to fetch the ClassificationBank object w/ ancestry
103   cbank_obj = @service.classificationbank_object( 
104                 :hierarchiesID => cbank_id, 
105                 :ancestryFlag => 1 )
106   
107   # walk UP the lineage creating new taxa if they don't exist, and
108   # stopping if we find one
109   lineage = [taxon]
110   cbank_obj.search('//ancestry/value').each do |ancestor|
111     namebank_id = ancestor.at('namebankID').inner_text
112     cbank_id = ancestor.at('classificationBankID').inner_text
113     begin
114       rdf = @service.lsid(:namespace => 'classificationbank', 
115                           :object => cbank_id)
116       new_taxon = UBioTaxonAdapter.new(rdf, :name_provider => self)
117     rescue TaxonAdapterError
118       # if the cbank object fails, try converting from namebank
119       rdf = @service.lsid(:namespace => 'namebank', 
120                           :object => namebank_id)
121       new_taxon = begin
122         UBioTaxonAdapter.new(rdf)
123       rescue TaxonAdapterError => e
124         raise NameProviderError, "uBio bonked: #{e}"
125       end
126     rescue UBioConnectionError => e
127       taxon.logger.error("Error while running get_lineage_for(#{taxon}): #{e}") if taxon.logger
128       raise NameProviderError, e.message
129     rescue StandardError => e
130       raise NameProviderError, "uBio bonked: #{e}"
131     end
132 
133     lineage << new_taxon
134   end
135 
136   lineage.compact
137 end

139 def get_phylum_for(taxon, lineage = nil)
140   # Try to avoid calling uBio a billion times using their
141   # taxonomicGroup element
142   if taxon.respond_to?(:hxml) && (taxonomic_group = taxon.hxml.at('//ubio:taxonomicGroup'))
143     if taxonomic_group_taxon = Taxon.find_by_name(taxonomic_group.inner_text)
144       return taxonomic_group_taxon if taxonomic_group_taxon.rank == 'phylum'
145       return taxonomic_group_taxon.phylum
146     end
147   end
148   
149   begin
150     lineage ||= get_lineage_for(taxon)
151   rescue NameProviderError => e
152     return nil
153   end
154   phylum = lineage.detect{|t| t.rank && t.rank.downcase == 'phylum'}
155   phylum ||= lineage.last.phylum
156   phylum
157 end

    # File lib/ratatosk/lib/ratatosk/name_providers/ubio.rb
166 def choose_cbank_id(cbank_response)
167   cbank_results = cbank_response.search('//seniorNames/value')
168   if cbank_results.empty?
169     raise NameProviderError, 
170           "uBio doesn't have any classification data for this taxon"
171   end
172 
173   cbank_title_ids = {}
174   cbank_results.each do |c|
175     cbank_title = Base64.decode64(c.at('classificationTitle').inner_text)
176     cbank_id = c.at('classificationBankID').inner_text
177     if cbank_title and cbank_id
178       cbank_title_ids[cbank_title] = cbank_id
179     end
180   end
181 
182   preferred_id = nil
183   @PREFERRED_CLASSIFICATIONS.each do |c|
184     break if preferred_id = cbank_title_ids[c]
185   end
186   if preferred_id.nil?
187     preferred = cbank_title_ids.select do |title, id|
188       not @REJECTED_CLASSIFICATIONS.include?(title)
189     end.first
190     if preferred.nil?
191       raise NameProviderError, 
192             "uBio only has classification data for this taxon " + 
193             "from incompatible classifications " + 
194             "(#{cbank_title_ids.keys.join(', ')})"
195     end
196     preferred_id = preferred.last
197   end
198   preferred_id
199 end
