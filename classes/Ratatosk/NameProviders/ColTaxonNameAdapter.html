    # File lib/ratatosk/lib/ratatosk/name_providers/catalogue_of_life.rb
139 def get_taxon
140   if is_accepted_sciname?
141     taxon = ColTaxonAdapter.new(@hxml)
142   else
143     taxon = ColTaxonAdapter.new(accepted_name_hxml)
144   end
145   
146   # This is necessary because calling save here runs the validations,
147   # sees that the Taxon is new and declares the lexion validation ok,
148   # then saves the new taxon, which would fire the after save callback
149   # creating another taxon name, and then this taxon name gets
150   # created, resulting in duplicate, invalid taxon names.
151   taxon.skip_new_taxon_name = true
152   
153   taxon
154 end
