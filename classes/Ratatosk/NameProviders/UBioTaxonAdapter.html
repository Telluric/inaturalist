240 def get_name
241   begin
242     name = @hxml.at('//ubio:canonicalName').inner_text
243   rescue NoMethodError, Nokogiri::XML::XPath::SyntaxError
244     begin
245       name = @hxml.at('//dc:title').inner_text
246     rescue NoMethodError, Nokogiri::XML::XPath::SyntaxError
247       # without any kind of name in the RDF response, we can't make a taxon
248       raise TaxonAdapterError, "Couldn't find a name in a uBio RDF response"
249     end
250   end
251   name
252 end

254 def get_namebank_id
255   # CBank and NBank store the namebank LSID in different places
256   if @hxml.at('//rdf:Description')['rdf:about'] =~ /classificationbank/
257     lsid = @hxml.at('//ubio:namebankIdentifier')['rdf:resource']
258   else
259     lsid = @hxml.at('//dc:identifier').inner_text
260   end
261 
262   namebank_id = lsid.split(':')[4] # 4th term should be the identifier
263   if namebank_id.nil?
264     raise TaxonAdapterError, 
265           "Couldn't find a valid LSID in uBio's RDF response"
266   end
267   namebank_id
268 end
