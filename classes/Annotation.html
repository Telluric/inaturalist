189 def self.reassess_annotations_for_taxon_id( taxon )
190   taxon = Taxon.find_by_id(taxon) unless taxon.is_a?(Taxon)
191   Annotation.
192       joins(
193         controlled_attribute: {
194           controlled_term_taxa: :taxon
195         }
196       ).
197       where( taxon.subtree_conditions ).
198       includes(
199         { resource: :taxon },
200         controlled_value: [
201           :excepted_taxa,
202           :taxa,

204         ],
205         controlled_attribute: [
206           :excepted_taxa,
207           :taxa,
208           :controlled_term_taxa
209         ]
210       ).
211       find_each do |a|
212     # run the validation methods which might be affected by taxon changes
213     a.attribute_belongs_to_taxon
214     a.value_belongs_to_taxon

216     a.destroy if a.errors.any?
