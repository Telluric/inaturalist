11 def index
12   @comments = Comment.includes(:user).order( "comments.id DESC" ).page( params[:page] )
13   if logged_in? && (!params[:mine].blank? || !params[:for_me].blank? || !params[:q].blank?)
14     filtering = true
15     if !params[:mine].blank?
16       @comments = @comments.by(current_user)
17     elsif !params[:for_me].blank?
18       @comments = @comments.for_observer(current_user)
19     end
20     @comments = @comments.dbsearch(params[:q]) unless params[:q].blank?
21   end
22   if !filtering && @site && @site.site_only_users
23     @comments = @comments.joins(:user).where("users.site_id = ?", @site)
24   end
25   parent_ids = @comments.map(&:parent_id)
26   @extra_comments = Comment.where( parent_id: parent_ids ).
27     where( "comments.id NOT IN (?)", @comments.map(&:id) ).includes(:user)
28   @extra_ids = Identification.where( observation_id: parent_ids ).includes(:observation)
29   comments_and_ids = [@comments, @extra_comments, @extra_ids].flatten.uniq
30   @comments_by_parent_id = comments_and_ids.sort_by(&:created_at).group_by do |c|
31     if c.respond_to? :observation_id
32       [c.observation.class.to_s, c.observation_id].join( "_" )
33     else
34       [c.parent.class.to_s, c.parent_id].join( "_" )
35     end
36   end
37   @latest_comments = @comments_by_parent_id.map {|g,items|
38     items.select {|i| i.is_a?( Comment ) }.compact.sort_by(&:created_at).last
39   }.compact.sort_by(&:created_at).reverse
40   respond_to do |format|
41     format.html do
42       if params[:partial]
43         render partial: "listing_for_dashboard",
44           collection: @latest_comments, layout: false
45       end
46     end
47   end
48 end
